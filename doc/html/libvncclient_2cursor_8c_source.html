42 rfbBool HandleCursorShape(rfbClient* client,int xhot, int yhot, int width, int height, uint32_t enc)
43 {
44  int bytesPerPixel;
45  size_t bytesPerRow, bytesMaskData;
46  rfbXCursorColors rgb;
47  uint32_t colors[2];
48  char *buf;
49  uint8_t *ptr;
50  int x, y, b;
51 
52  bytesPerPixel = client->format.bitsPerPixel / 8;
53  bytesPerRow = (width + 7) / 8;
54  bytesMaskData = bytesPerRow * height;
55 
56  if (width * height == 0)
57  return TRUE;
58 
59  if (width >= MAX_CURSOR_SIZE || height >= MAX_CURSOR_SIZE)
60  return FALSE;
61 
62  /* Allocate memory for pixel data and temporary mask data. */
63  if(client->rcSource)
64  free(client->rcSource);
65 
66  client->rcSource = malloc(width * height * bytesPerPixel);
67  if (client->rcSource == NULL)
68  return FALSE;
69 
70  buf = malloc(bytesMaskData);
71  if (buf == NULL) {
72  free(client->rcSource);
73  client->rcSource = NULL;
74  return FALSE;
75  }
76 
77  /* Read and decode cursor pixel data, depending on the encoding type. */
78 
79  if (enc == rfbEncodingXCursor) {
80  /* Read and convert background and foreground colors. */
81  if (!ReadFromRFBServer(client, (char *)&rgb, sz_rfbXCursorColors)) {
82  free(client->rcSource);
83  client->rcSource = NULL;
84  free(buf);
85  return FALSE;
86  }
87  colors[0] = RGB24_TO_PIXEL(32, rgb.backRed, rgb.backGreen, rgb.backBlue);
88  colors[1] = RGB24_TO_PIXEL(32, rgb.foreRed, rgb.foreGreen, rgb.foreBlue);
89 
90  /* Read 1bpp pixel data into a temporary buffer. */
91  if (!ReadFromRFBServer(client, buf, bytesMaskData)) {
92  free(client->rcSource);
93  client->rcSource = NULL;
94  free(buf);
95  return FALSE;
96  }
97 
98  /* Convert 1bpp data to byte-wide color indices. */
99  ptr = client->rcSource;
100  for (y = 0; y < height; y++) {
101  for (x = 0; x < width / 8; x++) {
102  for (b = 7; b >= 0; b--) {
103  *ptr = buf[y * bytesPerRow + x] >> b & 1;
104  ptr += bytesPerPixel;
105  }
106  }
107  for (b = 7; b > 7 - width % 8; b--) {
108  *ptr = buf[y * bytesPerRow + x] >> b & 1;
109  ptr += bytesPerPixel;
110  }
111  }
112 
113  /* Convert indices into the actual pixel values. */
114  switch (bytesPerPixel) {
115  case 1:
116  for (x = 0; x < width * height; x++)
117  client->rcSource[x] = (uint8_t)colors[client->rcSource[x]];
118  break;
119  case 2:
120  for (x = 0; x < width * height; x++)
121  ((uint16_t *)client->rcSource)[x] = (uint16_t)colors[client->rcSource[x * 2]];
122  break;
123  case 4:
124  for (x = 0; x < width * height; x++)
125  ((uint32_t *)client->rcSource)[x] = colors[client->rcSource[x * 4]];
126  break;
127  }
128 
129  } else { /* enc == rfbEncodingRichCursor */
130 
131  if (!ReadFromRFBServer(client, (char *)client->rcSource, width * height * bytesPerPixel)) {
132  free(client->rcSource);
133  client->rcSource = NULL;
134  free(buf);
135  return FALSE;
136  }
137 
138  }
139 
140  /* Read and decode mask data. */
141 
142  if (!ReadFromRFBServer(client, buf, bytesMaskData)) {
143  free(client->rcSource);
144  client->rcSource = NULL;
145  free(buf);
146  return FALSE;
147  }
148 
149  client->rcMask = malloc(width * height);
150  if (client->rcMask == NULL) {
151  free(client->rcSource);
152  client->rcSource = NULL;
153  free(buf);
154  return FALSE;
155  }
156 
157  ptr = client->rcMask;
158  for (y = 0; y < height; y++) {
159  for (x = 0; x < width / 8; x++) {
160  for (b = 7; b >= 0; b--) {
161  *ptr++ = buf[y * bytesPerRow + x] >> b & 1;
162  }
163  }
164  for (b = 7; b > 7 - width % 8; b--) {
165  *ptr++ = buf[y * bytesPerRow + x] >> b & 1;
166  }
167  }
168 
169  if (client->GotCursorShape != NULL) {
170  client->GotCursorShape(client, xhot, yhot, width, height, bytesPerPixel);
171  }
172 
173  free(buf);
174 
175  return TRUE;
176 }
