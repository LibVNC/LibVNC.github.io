34 static rfbBool
35 HandleUltraBPP (rfbClient* client, int rx, int ry, int rw, int rh)
36 {
37  rfbZlibHeader hdr;
38  int toRead=0;
39  int inflateResult=0;
40  lzo_uint uncompressedBytes = (( rw * rh ) * ( BPP / 8 ));
41 
42  if (!ReadFromRFBServer(client, (char *)&hdr, sz_rfbZlibHeader))
43  return FALSE;
44 
45  toRead = rfbClientSwap32IfLE(hdr.nBytes);
46  if (toRead==0) return TRUE;
47 
48  if (uncompressedBytes==0)
49  {
50  rfbClientLog("ultra error: rectangle has 0 uncomressed bytes ((%dw * %dh) * (%d / 8))\n", rw, rh, BPP);
51  return FALSE;
52  }
53 
54  /* First make sure we have a large enough raw buffer to hold the
55  * decompressed data. In practice, with a fixed BPP, fixed frame
56  * buffer size and the first update containing the entire frame
57  * buffer, this buffer allocation should only happen once, on the
58  * first update.
59  */
60  if ( client->raw_buffer_size < (int)uncompressedBytes) {
61  if ( client->raw_buffer != NULL ) {
62  free( client->raw_buffer );
63  }
64  client->raw_buffer_size = uncompressedBytes;
65  /* buffer needs to be aligned on 4-byte boundaries */
66  if ((client->raw_buffer_size % 4)!=0)
67  client->raw_buffer_size += (4-(client->raw_buffer_size % 4));
68  client->raw_buffer = (char*) malloc( client->raw_buffer_size );
69  if(client->raw_buffer == NULL)
70  return FALSE;
71  }
72 
73  /* allocate enough space to store the incoming compressed packet */
74  if ( client->ultra_buffer_size < toRead ) {
75  if ( client->ultra_buffer != NULL ) {
76  free( client->ultra_buffer );
77  }
78  client->ultra_buffer_size = toRead;
79  /* buffer needs to be aligned on 4-byte boundaries */
80  if ((client->ultra_buffer_size % 4)!=0)
81  client->ultra_buffer_size += (4-(client->ultra_buffer_size % 4));
82  client->ultra_buffer = (char*) malloc( client->ultra_buffer_size );
83  }
84 
85  /* Fill the buffer, obtaining data from the server. */
86  if (!ReadFromRFBServer(client, client->ultra_buffer, toRead))
87  return FALSE;
88 
89  /* uncompress the data */
90  uncompressedBytes = client->raw_buffer_size;
91  inflateResult = lzo1x_decompress_safe(
92  (lzo_byte *)client->ultra_buffer, toRead,
93  (lzo_byte *)client->raw_buffer, (lzo_uintp) &uncompressedBytes,
94  NULL);
95 
96  /* Note that uncompressedBytes will be 0 on output overrun */
97  if ((rw * rh * (BPP / 8)) != uncompressedBytes)
98  rfbClientLog("Ultra decompressed unexpected amount of data (%d != %d)\n", (rw * rh * (BPP / 8)), uncompressedBytes);
99 
100  /* Put the uncompressed contents of the update on the screen. */
101  if ( inflateResult == LZO_E_OK )
102  {
103  client->GotBitmap(client, (unsigned char *)client->raw_buffer, rx, ry, rw, rh);
104  }
105  else
106  {
107  rfbClientLog("ultra decompress returned error: %d\n",
108  inflateResult);
109  return FALSE;
110  }
111  return TRUE;
112 }

116 static rfbBool
117 HandleUltraZipBPP (rfbClient* client, int rx, int ry, int rw, int rh)
118 {
119  rfbZlibHeader hdr;
120  int i=0;
121  int toRead=0;
122  int inflateResult=0;
123  unsigned char *ptr=NULL;
124  lzo_uint uncompressedBytes = ry + (rw * 65535);
125  unsigned int numCacheRects = rx;
126 
127  if (!ReadFromRFBServer(client, (char *)&hdr, sz_rfbZlibHeader))
128  return FALSE;
129 
130  toRead = rfbClientSwap32IfLE(hdr.nBytes);
131 
132  if (toRead==0) return TRUE;
133 
134  if (uncompressedBytes==0)
135  {
136  rfbClientLog("ultrazip error: rectangle has 0 uncomressed bytes (%dy + (%dw * 65535)) (%d rectangles)\n", ry, rw, rx);
137  return FALSE;
138  }
139 
140  /* First make sure we have a large enough raw buffer to hold the
141  * decompressed data. In practice, with a fixed BPP, fixed frame
142  * buffer size and the first update containing the entire frame
143  * buffer, this buffer allocation should only happen once, on the
144  * first update.
145  */
146  if ( client->raw_buffer_size < (int)(uncompressedBytes + 500)) {
147  if ( client->raw_buffer != NULL ) {
148  free( client->raw_buffer );
149  }
150  client->raw_buffer_size = uncompressedBytes + 500;
151  /* buffer needs to be aligned on 4-byte boundaries */
152  if ((client->raw_buffer_size % 4)!=0)
153  client->raw_buffer_size += (4-(client->raw_buffer_size % 4));
154  client->raw_buffer = (char*) malloc( client->raw_buffer_size );
155  if(client->raw_buffer == NULL)
156  return FALSE;
157  }
158

161  if ( client->ultra_buffer_size < toRead ) {
162  if ( client->ultra_buffer != NULL ) {
163  free( client->ultra_buffer );
164  }
165  client->ultra_buffer_size = toRead;
166  client->ultra_buffer = (char*) malloc( client->ultra_buffer_size );
167  }
168 
169  /* Fill the buffer, obtaining data from the server. */
170  if (!ReadFromRFBServer(client, client->ultra_buffer, toRead))
171  return FALSE;
172 
173  /* uncompress the data */
174  uncompressedBytes = client->raw_buffer_size;
175  inflateResult = lzo1x_decompress_safe(
176  (lzo_byte *)client->ultra_buffer, toRead,
177  (lzo_byte *)client->raw_buffer, &uncompressedBytes, NULL);
178  if ( inflateResult != LZO_E_OK )
179  {
180  rfbClientLog("ultra decompress returned error: %d\n",
181  inflateResult);
182  return FALSE;
183  }
184 
185  /* Put the uncompressed contents of the update on the screen. */
186  ptr = (unsigned char *)client->raw_buffer;
187  for (i=0; i<numCacheRects; i++)
188  {
189  unsigned short sx, sy, sw, sh;
190  unsigned int se;
191 
192  memcpy((char *)&sx, ptr, 2); ptr += 2;
193  memcpy((char *)&sy, ptr, 2); ptr += 2;
194  memcpy((char *)&sw, ptr, 2); ptr += 2;
195  memcpy((char *)&sh, ptr, 2); ptr += 2;
196  memcpy((char *)&se, ptr, 4); ptr += 4;
197 
198  sx = rfbClientSwap16IfLE(sx);
199  sy = rfbClientSwap16IfLE(sy);
200  sw = rfbClientSwap16IfLE(sw);
201  sh = rfbClientSwap16IfLE(sh);
202  se = rfbClientSwap32IfLE(se);
203 
204  if (se == rfbEncodingRaw)
205  {
206  client->GotBitmap(client, (unsigned char *)ptr, sx, sy, sw, sh);
207  ptr += ((sw * sh) * (BPP / 8));
208  }
209  }
210 
211  return TRUE;
212 }
