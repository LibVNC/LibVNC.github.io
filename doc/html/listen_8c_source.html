44 /*
45  * listenForIncomingConnections() - listen for incoming connections from
46  * servers, and fork a new process to deal with each connection.
47  */
48 
49 void
50 listenForIncomingConnections(rfbClient* client)
51 {
52 #ifdef WIN32
53  /* FIXME */
54  rfbClientErr("listenForIncomingConnections on MinGW32 NOT IMPLEMENTED\n");
55  return;
56 #else
57  int listenSocket, listen6Socket = -1;
58  fd_set fds;
59 
60  client->listenSpecified = TRUE;
61 
62  listenSocket = ListenAtTcpPortAndAddress(client->listenPort, client->listenAddress);
63 
64  if ((listenSocket < 0))
65  return;
66 
67  rfbClientLog("%s -listen: Listening on port %d\n",
68  client->programName,client->listenPort);
69  rfbClientLog("%s -listen: Command line errors are not reported until "
70  "a connection comes in.\n", client->programName);
71 
72 #ifdef LIBVNCSERVER_IPv6 /* only try that if we're IPv6-capable, otherwise we may try to bind to the same port which would make all that listening fail */
73  /* only do IPv6 listen of listen6Port is set */
74  if (client->listen6Port > 0)
75  {
76  listen6Socket = ListenAtTcpPortAndAddress(client->listen6Port, client->listen6Address);
77 
78  if (listen6Socket < 0)
79  return;
80 
81  rfbClientLog("%s -listen: Listening on IPV6 port %d\n",
82  client->programName,client->listenPort);
83  rfbClientLog("%s -listen: Command line errors are not reported until "
84  "a connection comes in.\n", client->programName);
85  }
86 #endif
87 
88  while (TRUE) {
89  int r;
90  /* reap any zombies */
91  int status, pid;
92  while ((pid= wait4(-1, &status, WNOHANG, (struct rusage *)0))>0);
93 
94  /* TODO: callback for discard any events (like X11 events) */
95 
96  FD_ZERO(&fds);
97 
98  if(listenSocket >= 0)
99  FD_SET(listenSocket, &fds);
100  if(listen6Socket >= 0)
101  FD_SET(listen6Socket, &fds);
102 
103  r = select(rfbMax(listenSocket, listen6Socket)+1, &fds, NULL, NULL, NULL);
104 
105  if (r > 0) {
106  if (FD_ISSET(listenSocket, &fds))
107  client->sock = AcceptTcpConnection(client->listenSock);
108  else if (FD_ISSET(listen6Socket, &fds))
109  client->sock = AcceptTcpConnection(client->listen6Sock);
110 
111  if (client->sock < 0)
112  return;
113  if (!SetNonBlocking(client->sock))
114  return;
115 
116  /* Now fork off a new process to deal with it... */
117 
118  switch (fork()) {
119 
120  case -1:
121  rfbClientErr("fork\n");
122  return;
123 
124  case 0:
125  /* child - return to caller */
126  close(listenSocket);
127  close(listen6Socket);
128  return;
129 
130  default:
131  /* parent - go round and listen again */
132  close(client->sock);
133  break;
134  }
135  }
136  }
137 #endif
138 }

142 /*
143  * listenForIncomingConnectionsNoFork() - listen for incoming connections
144  * from servers, but DON'T fork, instead just wait timeout microseconds.
145  * If timeout is negative, block indefinitely.
146  * Returns 1 on success (there was an incoming connection on the listen socket
147  * and we accepted it successfully), -1 on error, 0 on timeout.
148  */
149 
150 int
151 listenForIncomingConnectionsNoFork(rfbClient* client, int timeout)
152 {
153  fd_set fds;
154  struct timeval to;
155  int r;
156 
157  to.tv_sec= timeout / 1000000;
158  to.tv_usec= timeout % 1000000;
159 
160  client->listenSpecified = TRUE;
161 
162  if (client->listenSock < 0)
163  {
164  client->listenSock = ListenAtTcpPortAndAddress(client->listenPort, client->listenAddress);
165 
166  if (client->listenSock < 0)
167  return -1;
168 
169  rfbClientLog("%s -listennofork: Listening on port %d\n",
170  client->programName,client->listenPort);
171  rfbClientLog("%s -listennofork: Command line errors are not reported until "
172  "a connection comes in.\n", client->programName);
173  }
174 
175 #ifdef LIBVNCSERVER_IPv6 /* only try that if we're IPv6-capable, otherwise we may try to bind to the same port which would make all that listening fail */
176  /* only do IPv6 listen of listen6Port is set */
177  if (client->listen6Port > 0 && client->listen6Sock < 0)
178  {
179  client->listen6Sock = ListenAtTcpPortAndAddress(client->listen6Port, client->listen6Address);
180 
181  if (client->listen6Sock < 0)
182  return -1;
183 
184  rfbClientLog("%s -listennofork: Listening on IPV6 port %d\n",
185  client->programName,client->listenPort);
186  rfbClientLog("%s -listennofork: Command line errors are not reported until "
187  "a connection comes in.\n", client->programName);
188  }
189 #endif
190 
191  FD_ZERO(&fds);
192 
193  if(client->listenSock >= 0)
194  FD_SET(client->listenSock, &fds);
195  if(client->listen6Sock >= 0)
196  FD_SET(client->listen6Sock, &fds);
197 
198  if (timeout < 0)
199  r = select(rfbMax(client->listenSock, client->listen6Sock) +1, &fds, NULL, NULL, NULL);
200  else
201  r = select(rfbMax(client->listenSock, client->listen6Sock) +1, &fds, NULL, NULL, &to);
202 
203  if (r > 0)
204  {
205  if (FD_ISSET(client->listenSock, &fds))
206  client->sock = AcceptTcpConnection(client->listenSock);
207  else if (FD_ISSET(client->listen6Sock, &fds))
208  client->sock = AcceptTcpConnection(client->listen6Sock);
209 
210  if (client->sock < 0)
211  return -1;
212  if (!SetNonBlocking(client->sock))
213  return -1;
214 
215  if(client->listenSock >= 0) {
216  close(client->listenSock);
217  client->listenSock = -1;
218  }
219  if(client->listen6Sock >= 0) {
220  close(client->listen6Sock);
221  client->listen6Sock = -1;
222  }
223  return r;
224  }
225 
226  /* r is now either 0 (timeout) or -1 (error) */
227  return r;
228 }
