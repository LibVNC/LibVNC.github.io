362  ws_dbg("bufsize=%d\n", bufsize);

365  } else {

408  * the whole frame is received and carry over any remaining bytes in the carry buf*/

410

413  memcpy(&tmp, data + i * sizeof(tmp), sizeof(tmp));
414  tmp ^= wsctx->header.mask.u;
415  memcpy(data + i * sizeof(tmp), &tmp, sizeof(tmp));
416  }

418

421  for (i*=4; i < toDecode; i++) {
422  data[i] ^= wsctx->header.mask.c[i % 4];
423  }
424

426  wsctx->carrylen = 0;
427  } else {

429  wsctx->carrylen = toDecode - (i * 4);
430  if (wsctx->carrylen < 0 || wsctx->carrylen > ARRAYSIZE(wsctx->carryBuf)) {
431  rfbErr("%s: internal error, invalid carry over size: carrylen=%d, toDecode=%d, i=%d", __func__, wsctx->carrylen, toDecode, i);
432  *sockRet = -1;
433  errno = EIO;
434  return WS_HYBI_STATE_ERR;
435  }

437  memcpy(wsctx->carryBuf, data + (i * 4), wsctx->carrylen);
438  wsctx->writePos -= wsctx->carrylen;
439  }
