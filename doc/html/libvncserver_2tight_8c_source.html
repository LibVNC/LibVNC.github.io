291 rfbBool
292 SendRectEncodingTight(rfbClientPtr cl,
293  int x,
294  int y,
295  int w,
296  int h)
297 {
298  int nMaxRows;
299  uint32_t colorValue;
300  int dx, dy, dw, dh;
301  int x_best, y_best, w_best, h_best;
302  char *fbptr;
303 
304  rfbSendUpdateBuf(cl);
305 
306  compressLevel = cl->tightCompressLevel;
307  qualityLevel = cl->turboQualityLevel;
308  subsampLevel = cl->turboSubsampLevel;
309 
310  /* We only allow compression levels that have a demonstrable performance
311  benefit. CL 0 with JPEG reduces CPU usage for workloads that have low
312  numbers of unique colors, but the same thing can be accomplished by
313  using CL 0 without JPEG (AKA "Lossless Tight.") For those same
314  low-color workloads, CL 2 can provide typically 20-40% better
315  compression than CL 1 (with a commensurate increase in CPU usage.) For
316  high-color workloads, CL 1 should always be used, as higher compression
317  levels increase CPU usage for these workloads without providing any
318  significant reduction in bandwidth. */
319  if (qualityLevel != -1) {
320  if (compressLevel < 1) compressLevel = 1;
321  if (compressLevel > 2) compressLevel = 2;
322  }
323 
324  /* With JPEG disabled, CL 2 offers no significant bandwidth savings over
325  CL 1, so we don't include it. */
326  else if (compressLevel > 1) compressLevel = 1;
327 
328  /* CL 9 (which maps internally to CL 3) is included mainly for backward
329  compatibility with TightVNC Compression Levels 5-9. It should be used
330  only in extremely low-bandwidth cases in which it can be shown to have a
331  benefit. For low-color workloads, it provides typically only 10-20%
332  better compression than CL 2 with JPEG and CL 1 without JPEG, and it
333  uses, on average, twice as much CPU time. */
334  if (cl->tightCompressLevel == 9) compressLevel = 3;
335 
336  if ( cl->format.depth == 24 && cl->format.redMax == 0xFF &&
337  cl->format.greenMax == 0xFF && cl->format.blueMax == 0xFF ) {
338  usePixelFormat24 = TRUE;
339  } else {
340  usePixelFormat24 = FALSE;
341  }
342 
343  if (!cl->enableLastRectEncoding || w * h < MIN_SPLIT_RECT_SIZE)
344  return SendRectSimple(cl, x, y, w, h);
345 
346  /* Make sure we can write at least one pixel into tightBeforeBuf. */
347 
348  if (tightBeforeBufSize < 4) {
349  tightBeforeBufSize = 4;
350  if (tightBeforeBuf == NULL)
351  tightBeforeBuf = (char *)malloc(tightBeforeBufSize);
352  else
353  tightBeforeBuf = (char *)realloc(tightBeforeBuf,
354  tightBeforeBufSize);
355  }
356 
357  /* Calculate maximum number of rows in one non-solid rectangle. */
358 
359  {
360  int maxRectSize, maxRectWidth, nMaxWidth;
361 
362  maxRectSize = tightConf[compressLevel].maxRectSize;
363  maxRectWidth = tightConf[compressLevel].maxRectWidth;
364  nMaxWidth = (w > maxRectWidth) ? maxRectWidth : w;
365  nMaxRows = maxRectSize / nMaxWidth;
366  }
367 
368  /* Try to find large solid-color areas and send them separately. */
369 
370  for (dy = y; dy < y + h; dy += MAX_SPLIT_TILE_SIZE) {
371 
372  /* If a rectangle becomes too large, send its upper part now. */
373 
374  if (dy - y >= nMaxRows) {
375  if (!SendRectSimple(cl, x, y, w, nMaxRows))
376  return 0;
377  y += nMaxRows;
378  h -= nMaxRows;
379  }
380 
381  dh = (dy + MAX_SPLIT_TILE_SIZE <= y + h) ?
382  MAX_SPLIT_TILE_SIZE : (y + h - dy);
383 
384  for (dx = x; dx < x + w; dx += MAX_SPLIT_TILE_SIZE) {
385 
386  dw = (dx + MAX_SPLIT_TILE_SIZE <= x + w) ?
387  MAX_SPLIT_TILE_SIZE : (x + w - dx);
388 
389  if (CheckSolidTile(cl, dx, dy, dw, dh, &colorValue, FALSE)) {
390 
391  if (subsampLevel == TJ_GRAYSCALE && qualityLevel != -1) {
392  uint32_t r = (colorValue >> 16) & 0xFF;
393  uint32_t g = (colorValue >> 8) & 0xFF;
394  uint32_t b = (colorValue) & 0xFF;
395  double y = (0.257 * (double)r) + (0.504 * (double)g)
396  + (0.098 * (double)b) + 16.;
397  colorValue = (int)y + (((int)y) << 8) + (((int)y) << 16);
398  }
399 
400  /* Get dimensions of solid-color area. */
401 
402  FindBestSolidArea(cl, dx, dy, w - (dx - x), h - (dy - y),
403  colorValue, &w_best, &h_best);
404 
405  /* Make sure a solid rectangle is large enough
406  (or the whole rectangle is of the same color). */
407 
408  if ( w_best * h_best != w * h &&
409  w_best * h_best < MIN_SOLID_SUBRECT_SIZE )
410  continue;
411 
412  /* Try to extend solid rectangle to maximum size. */
413 
414  x_best = dx; y_best = dy;
415  ExtendSolidArea(cl, x, y, w, h, colorValue,
416  &x_best, &y_best, &w_best, &h_best);
417 
418  /* Send rectangles at top and left to solid-color area. */
419 
420  if ( y_best != y &&
421  !SendRectSimple(cl, x, y, w, y_best-y) )
422  return FALSE;
423  if ( x_best != x &&
424  !SendRectEncodingTight(cl, x, y_best,
425  x_best-x, h_best) )
426  return FALSE;
427 
428  /* Send solid-color rectangle. */
429 
430  if (!rfbSendTightHeader(cl, x_best, y_best, w_best, h_best))
431  return FALSE;
432 
433  fbptr = (cl->scaledScreen->frameBuffer +
434  (cl->scaledScreen->paddedWidthInBytes * y_best) +
435  (x_best * (cl->scaledScreen->bitsPerPixel / 8)));
436 
437  (*cl->translateFn)(cl->translateLookupTable, &cl->screen->serverFormat,
438  &cl->format, fbptr, tightBeforeBuf,
439  cl->scaledScreen->paddedWidthInBytes, 1, 1);
440 
441  if (!SendSolidRect(cl))
442  return FALSE;
443 
444  /* Send remaining rectangles (at right and bottom). */
445 
446  if ( x_best + w_best != x + w &&
447  !SendRectEncodingTight(cl, x_best + w_best, y_best,
448  w - (x_best-x) - w_best, h_best) )
449  return FALSE;
450  if ( y_best + h_best != y + h &&
451  !SendRectEncodingTight(cl, x, y_best + h_best,
452  w, h - (y_best-y) - h_best) )
453  return FALSE;
454 
455  /* Return after all recursive calls are done. */
456 
457  return TRUE;
458  }
459 
460  }
461 
462  }
463 
464  /* No suitable solid-color rectangles found. */
465 
466  return SendRectSimple(cl, x, y, w, h);
467 }

470 static void
471 FindBestSolidArea(rfbClientPtr cl,
472  int x,
473  int y,
474  int w,
475  int h,
476  uint32_t colorValue,
477  int *w_ptr,
478  int *h_ptr)
479 {
480  int dx, dy, dw, dh;
481  int w_prev;
482  int w_best = 0, h_best = 0;
483 
484  w_prev = w;
485 
486  for (dy = y; dy < y + h; dy += MAX_SPLIT_TILE_SIZE) {
487 
488  dh = (dy + MAX_SPLIT_TILE_SIZE <= y + h) ?
489  MAX_SPLIT_TILE_SIZE : (y + h - dy);
490  dw = (w_prev > MAX_SPLIT_TILE_SIZE) ?
491  MAX_SPLIT_TILE_SIZE : w_prev;
492 
493  if (!CheckSolidTile(cl, x, dy, dw, dh, &colorValue, TRUE))
494  break;
495 
496  for (dx = x + dw; dx < x + w_prev;) {
497  dw = (dx + MAX_SPLIT_TILE_SIZE <= x + w_prev) ?
498  MAX_SPLIT_TILE_SIZE : (x + w_prev - dx);
499  if (!CheckSolidTile(cl, dx, dy, dw, dh, &colorValue, TRUE))
500  break;
501  dx += dw;
502  }
503 
504  w_prev = dx - x;
505  if (w_prev * (dy + dh - y) > w_best * h_best) {
506  w_best = w_prev;
507  h_best = dy + dh - y;
508  }
509  }
510 
511  *w_ptr = w_best;
512  *h_ptr = h_best;
513 }
