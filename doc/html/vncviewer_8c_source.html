86 static rfbBool MallocFrameBuffer(rfbClient* client) {
87  uint64_t allocSize;
88 
89  if(client->frameBuffer)
90  free(client->frameBuffer);
91 
92  /* SECURITY: promote 'width' into uint64_t so that the multiplication does not overflow
93  'width' and 'height' are 16-bit integers per RFB protocol design
94  SIZE_MAX is the maximum value that can fit into size_t
95  */
96  allocSize = (uint64_t)client->width * client->height * client->format.bitsPerPixel/8;
97 
98  if (allocSize >= SIZE_MAX) {
99  rfbClientErr("CRITICAL: cannot allocate frameBuffer, requested size is too large\n");
100  return FALSE;
101  }
102 
103  client->frameBuffer=malloc( (size_t)allocSize );
104 
105  if (client->frameBuffer == NULL)
106  rfbClientErr("CRITICAL: frameBuffer allocation failed, requested size too large or not enough memory?\n");
107 
108  return client->frameBuffer?TRUE:FALSE;
109 }

117 static void FillRectangle(rfbClient* client, int x, int y, int w, int h, uint32_t colour) {
118  int i,j;
119 
120  if (client->frameBuffer == NULL) {
121  return;
122  }
123 
124  if (!CheckRect(client, x, y, w, h)) {
125  rfbClientLog("Rect out of bounds: %dx%d at (%d, %d)\n", x, y, w, h);
126  return;
127  }
128 
129 #define FILL_RECT(BPP) \
130  for(j=y*client->width;j<(y+h)*client->width;j+=client->width) \
131  for(i=x;i<x+w;i++) \
132  ((uint##BPP##_t*)client->frameBuffer)[j+i]=colour;
133 
134  switch(client->format.bitsPerPixel) {
135  case 8: FILL_RECT(8); break;
136  case 16: FILL_RECT(16); break;
137  case 32: FILL_RECT(32); break;
138  default:
139  rfbClientLog("Unsupported bitsPerPixel: %d\n",client->format.bitsPerPixel);
140  }

369 static rfbBool rfbInitConnection(rfbClient* client)
370 {
371  /* Unless we accepted an incoming connection, make a TCP connection to the
372  given VNC server */
373 
374  if (!client->listenSpecified) {
375  if (!client->serverHost)
376  return FALSE;
377  if (client->destHost) {
378  if (!ConnectToRFBRepeater(client,client->serverHost,client->serverPort,client->destHost,client->destPort))
379  return FALSE;
380  } else {
381  if (!ConnectToRFBServer(client,client->serverHost,client->serverPort))
382  return FALSE;
383  }
384  }
385 
386  /* Initialise the VNC connection, including reading the password */
387 
388  if (!InitialiseRFBConnection(client))
389  return FALSE;
390 
391  client->width=client->si.framebufferWidth;
392  client->height=client->si.framebufferHeight;
393  if (!client->MallocFrameBuffer(client))
394  return FALSE;
395 
396  if (!SetFormatAndEncodings(client))
397  return FALSE;
398 
399  if (client->updateRect.x < 0) {
400  client->updateRect.x = client->updateRect.y = 0;
401  client->updateRect.w = client->width;
402  client->updateRect.h = client->height;
403  }
404 
405  if (client->appData.scaleSetting>1)
406  {
407  if (!SendScaleSetting(client, client->appData.scaleSetting))
408  return FALSE;
409  if (!SendFramebufferUpdateRequest(client,
410  client->updateRect.x / client->appData.scaleSetting,
411  client->updateRect.y / client->appData.scaleSetting,
412  client->updateRect.w / client->appData.scaleSetting,
413  client->updateRect.h / client->appData.scaleSetting,
414  FALSE))
415  return FALSE;
416  }
417  else
418  {
419  if (!SendFramebufferUpdateRequest(client,
420  client->updateRect.x, client->updateRect.y,
421  client->updateRect.w, client->updateRect.h,
422  FALSE))
423  return FALSE;
424  }
425 
426  return TRUE;
427 }

437  j = i;

444  } else if (strcmp(argv[i], "-play") == 0) {
445  client->serverPort = -1;
446  j++;
447  } else if (i+1<*argc && strcmp(argv[i], "-encodings") == 0) {
448  client->appData.encodingsString = argv[i+1];
449  j+=2;
450  } else if (i+1<*argc && strcmp(argv[i], "-compress") == 0) {

452  j+=2;
453  } else if (i+1<*argc && strcmp(argv[i], "-quality") == 0) {
454  client->appData.qualityLevel = atoi(argv[i+1]);
455  j+=2;
456  } else if (i+1<*argc && strcmp(argv[i], "-scale") == 0) {

458  j+=2;
459  } else if (i+1<*argc && strcmp(argv[i], "-qosdscp") == 0) {
460  client->QoS_DSCP = atoi(argv[i+1]);
461  j+=2;
462  } else if (i+1<*argc && strcmp(argv[i], "-repeaterdest") == 0) {

464

472  client->destPort = atoi(colon+1);

476  char* colon=strrchr(argv[i],':');
477 
478  if(client->serverHost)

480 
481  if(colon) {

496  }
