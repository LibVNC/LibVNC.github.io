293 rfbBool
294 ConnectToRFBServer(rfbClient* client,const char *hostname, int port)
295 {
296  if (client->serverPort==-1) {
297  /* serverHost is a file recorded by vncrec. */
298  const char* magic="vncLog0.0";
299  char buffer[10];
300  rfbVNCRec* rec = (rfbVNCRec*)malloc(sizeof(rfbVNCRec));
301  client->vncRec = rec;
302 
303  rec->file = fopen(client->serverHost,"rb");
304  rec->tv.tv_sec = 0;
305  rec->readTimestamp = FALSE;
306  rec->doNotSleep = FALSE;
307 
308  if (!rec->file) {
309  rfbClientLog("Could not open %s.\n",client->serverHost);
310  return FALSE;
311  }
312  setbuf(rec->file,NULL);
313 
314  if (fread(buffer,1,strlen(magic),rec->file) != strlen(magic) || strncmp(buffer,magic,strlen(magic))) {
315  rfbClientLog("File %s was not recorded by vncrec.\n",client->serverHost);
316  fclose(rec->file);
317  return FALSE;
318  }
319  client->sock = -1;
320  return TRUE;
321  }
322 
323 #ifndef WIN32
324  if(IsUnixSocket(hostname))
325  /* serverHost is a UNIX socket. */
326  client->sock = ConnectClientToUnixSock(hostname);
327  else
328 #endif
329  {
330 #ifdef LIBVNCSERVER_IPv6
331  client->sock = ConnectClientToTcpAddr6(hostname, port);
332  if (client->sock == -1)
333 #endif
334  {
335  unsigned int host;
336 
337  /* serverHost is a hostname */
338  if (!StringToIPAddr(hostname, &host)) {
339  rfbClientLog("Couldn't convert '%s' to host address\n", hostname);
340  return FALSE;
341  }
342  client->sock = ConnectClientToTcpAddr(host, port);
343  }
344  }
345 
346  if (client->sock < 0) {
347  rfbClientLog("Unable to connect to VNC server\n");
348  return FALSE;
349  }
350 
351  if(client->QoS_DSCP && !SetDSCP(client->sock, client->QoS_DSCP))
352  return FALSE;
353 
354  return SetNonBlocking(client->sock);
355 }

368 #ifdef LIBVNCSERVER_IPv6

373  unsigned int host;

375  rfbClientLog("Couldn't convert '%s' to host address\n", repeaterHost);
376  return FALSE;
377  }
378 
379  client->sock = ConnectClientToTcpAddr(host, repeaterPort);
380  }
381

384  return FALSE;
385  }
386 
387  if (!SetNonBlocking(client->sock))

389

391  return FALSE;

393 
394  /* UltraVNC repeater always report version 000.000 to identify itself */
395  if (sscanf(pv,rfbProtocolVersionFormat,&major,&minor) != 2 || major != 0 || minor != 0) {
396  rfbClientLog("Not a valid VNC repeater (%s)\n",pv);
397  return FALSE;
398  }
399 
400  rfbClientLog("Connected to VNC repeater, using protocol version %d.%d\n", major, minor);
401 
402  tmphostlen = snprintf(tmphost, sizeof(tmphost), "%s:%d", destHost, destPort);
403  if(tmphostlen < 0 || tmphostlen >= (int)sizeof(tmphost))
404  return FALSE; /* snprintf error or output truncated */
405 
406  if (!WriteToRFBServer(client, tmphost, tmphostlen + 1))
407  return FALSE;
408 
409  return TRUE;
410 }

415 rfbBool
416 rfbHandleAuthResult(rfbClient* client)
417 {
418  uint32_t authResult=0, reasonLen=0;
419  char *reason=NULL;
420 
421  if (!ReadFromRFBServer(client, (char *)&authResult, 4)) return FALSE;
422 
423  authResult = rfbClientSwap32IfLE(authResult);
424 
425  switch (authResult) {
426  case rfbVncAuthOK:
427  rfbClientLog("VNC authentication succeeded\n");
428  return TRUE;
429  break;
430  case rfbVncAuthFailed:
431  if (client->major==3 && client->minor>7)
432  {
433  /* we have an error following */
434  if (!ReadFromRFBServer(client, (char *)&reasonLen, 4)) return FALSE;
435  reasonLen = rfbClientSwap32IfLE(reasonLen);
436  reason = malloc((uint64_t)reasonLen+1);
437  if (!ReadFromRFBServer(client, reason, reasonLen)) { free(reason); return FALSE; }
438  reason[reasonLen]=0;
439  rfbClientLog("VNC connection failed: %s\n",reason);
440  free(reason);
441  return FALSE;
442  }
443  rfbClientLog("VNC authentication failed\n");
444  return FALSE;
445  case rfbVncAuthTooMany:
446  rfbClientLog("VNC authentication failed - too many tries\n");
447  return FALSE;
448  }
449 
450  rfbClientLog("Unknown VNC authentication result: %d\n",
451  (int)authResult);
452  return FALSE;
453 }

471 static rfbBool
472 ReadSupportedSecurityType(rfbClient* client, uint32_t *result, rfbBool subAuth)
473 {
474  uint8_t count=0;
475  uint8_t loop=0;
476  uint8_t flag=0;
477  uint8_t tAuth[256];
478  char buf1[500],buf2[10];
479  uint32_t authScheme;
480 
481  if (!ReadFromRFBServer(client, (char *)&count, 1)) return FALSE;
482 
483  if (count==0)
484  {
485  rfbClientLog("List of security types is ZERO, expecting an error to follow\n");
486  ReadReason(client);
487  return FALSE;
488  }
489 
490  rfbClientLog("We have %d security types to read\n", count);
491  authScheme=0;
492  /* now, we have a list of available security types to read ( uint8_t[] ) */
493  for (loop=0;loop<count;loop++)
494  {
495  if (!ReadFromRFBServer(client, (char *)&tAuth[loop], 1)) return FALSE;
496  rfbClientLog("%d) Received security type %d\n", loop, tAuth[loop]);
497  if (flag) continue;
498  if (tAuth[loop]==rfbVncAuth || tAuth[loop]==rfbNoAuth ||
499 #if defined(LIBVNCSERVER_HAVE_GNUTLS) || defined(LIBVNCSERVER_HAVE_LIBSSL)
500  tAuth[loop]==rfbVeNCrypt ||
501 #endif
502 #ifdef LIBVNCSERVER_HAVE_SASL
503  tAuth[loop]==rfbSASL ||
504 #endif /* LIBVNCSERVER_HAVE_SASL */
505  (tAuth[loop]==rfbARD && client->GetCredential) ||
506  (!subAuth && (tAuth[loop]==rfbTLS || (tAuth[loop]==rfbVeNCrypt && client->GetCredential))))
507  {
508  if (!subAuth && client->clientAuthSchemes)
509  {
510  int i;
511  for (i=0;client->clientAuthSchemes[i];i++)
512  {
513  if (client->clientAuthSchemes[i]==(uint32_t)tAuth[loop])
514  {
515  flag++;
516  authScheme=tAuth[loop];
517  break;
518  }
519  }
520  }
521  else
522  {
523  flag++;
524  authScheme=tAuth[loop];
525  }
526  if (flag)
527  {
528  rfbClientLog("Selecting security type %d (%d/%d in the list)\n", authScheme, loop, count);
529  /* send back a single byte indicating which security type to use */
530  if (!WriteToRFBServer(client, (char *)&tAuth[loop], 1)) return FALSE;
531  }
532  }
533  }
534  if (authScheme==0)
535  {
536  memset(buf1, 0, sizeof(buf1));
537  for (loop=0;loop<count;loop++)
538  {
539  if (strlen(buf1)>=sizeof(buf1)-1) break;
540  snprintf(buf2, sizeof(buf2), (loop>0 ? ", %d" : "%d"), (int)tAuth[loop]);
541  strncat(buf1, buf2, sizeof(buf1)-strlen(buf1)-1);
542  }
543  rfbClientLog("Unknown authentication scheme from VNC server: %s\n",
544  buf1);
545  return FALSE;
546  }
547  *result = authScheme;
548  return TRUE;
549 }

551 static rfbBool
552 HandleVncAuth(rfbClient *client)
553 {
554  uint8_t challenge[CHALLENGESIZE];
555  char *passwd=NULL;
556  int i;
557 
558  if (!ReadFromRFBServer(client, (char *)challenge, CHALLENGESIZE)) return FALSE;
559 
560  if (client->serverPort!=-1) { /* if not playing a vncrec file */
561  if (client->GetPassword)
562  passwd = client->GetPassword(client);
563 
564  if ((!passwd) || (strlen(passwd) == 0)) {
565  rfbClientLog("Reading password failed\n");
566  return FALSE;
567  }
568  if (strlen(passwd) > 8) {
569  passwd[8] = '\0';
570  }
571 
572  rfbClientEncryptBytes(challenge, passwd);
573 
574  /* Lose the password from memory */
575  for (i = strlen(passwd); i >= 0; i--) {
576  passwd[i] = '\0';
577  }
578  free(passwd);
579 
580  if (!WriteToRFBServer(client, (char *)challenge, CHALLENGESIZE)) return FALSE;
581  }
582 
583  /* Handle the SecurityResult message */
584  if (!rfbHandleAuthResult(client)) return FALSE;
585 
586  return TRUE;
587 }

597 static rfbBool
598 HandlePlainAuth(rfbClient *client)
599 {
600  uint32_t ulen, ulensw;
601  uint32_t plen, plensw;
602  rfbCredential *cred;
603 
604  if (!client->GetCredential)
605  {
606  rfbClientLog("GetCredential callback is not set.\n");
607  return FALSE;
608  }
609  cred = client->GetCredential(client, rfbCredentialTypeUser);
610  if (!cred)
611  {
612  rfbClientLog("Reading credential failed\n");
613  return FALSE;
614  }
615 
616  ulen = (cred->userCredential.username ? strlen(cred->userCredential.username) : 0);
617  ulensw = rfbClientSwap32IfLE(ulen);
618  plen = (cred->userCredential.password ? strlen(cred->userCredential.password) : 0);
619  plensw = rfbClientSwap32IfLE(plen);
620  if (!WriteToRFBServer(client, (char *)&ulensw, 4) ||
621  !WriteToRFBServer(client, (char *)&plensw, 4))
622  {
623  FreeUserCredential(cred);
624  return FALSE;
625  }
626  if (ulen > 0)
627  {
628  if (!WriteToRFBServer(client, cred->userCredential.username, ulen))
629  {
630  FreeUserCredential(cred);
631  return FALSE;
632  }
633  }
634  if (plen > 0)
635  {
636  if (!WriteToRFBServer(client, cred->userCredential.password, plen))
637  {
638  FreeUserCredential(cred);
639  return FALSE;
640  }
641  }
642 
643  FreeUserCredential(cred);

651 /* Simple 64bit big integer arithmetic implementation */
652 /* (x + y) % m, works even if (x + y) > 64bit */
653 #define rfbAddM64(x,y,m) ((x+y)%m+(x+y<x?(((uint64_t)-1)%m+1)%m:0))
654 /* (x * y) % m */
655 static uint64_t
656 rfbMulM64(uint64_t x, uint64_t y, uint64_t m)
657 {
658  uint64_t r;
659  for(r=0;x>0;x>>=1)
660  {
661  if (x&1) r=rfbAddM64(r,y,m);
662  y=rfbAddM64(y,y,m);
663  }
664  return r;
665 }
666 /* (x ^ y) % m */
667 static uint64_t
668 rfbPowM64(uint64_t b, uint64_t e, uint64_t m)
669 {
670  uint64_t r;
671  for(r=1;e>0;e>>=1)
672  {
673  if(e&1) r=rfbMulM64(r,b,m);
674  b=rfbMulM64(b,b,m);
675  }
676  return r;
677 }

756 static rfbBool
757 HandleARDAuth(rfbClient *client)
758 {
759  uint8_t gen[2], len[2];
760  size_t keylen;
761  uint8_t *mod = NULL, *resp, *pub, *key, *shared;
762  gcry_mpi_t genmpi = NULL, modmpi = NULL, respmpi = NULL;
763  gcry_mpi_t privmpi = NULL, pubmpi = NULL, keympi = NULL;
764  gcry_md_hd_t md5 = NULL;
765  gcry_cipher_hd_t aes = NULL;
766  gcry_error_t error;
767  uint8_t userpass[128], ciphertext[128];

770  rfbBool result = FALSE;
771

773  {
774  /* Application did not initialize gcrypt, so we should */
775  if (!gcry_check_version(GCRYPT_VERSION))
776  {
777  /* Older version of libgcrypt is installed on system than compiled against */
778  rfbClientLog("libgcrypt version mismatch.\n");
779  }
780  }
781 
782  while (1)
783  {
784  if (!ReadFromRFBServer(client, (char *)gen, 2))
785  break;
786  if (!ReadFromRFBServer(client, (char *)len, 2))
787  break;

789  if (!client->GetCredential)
790  {
791  rfbClientLog("GetCredential callback is not set.\n");
792  break;
793  }
794  cred = client->GetCredential(client, rfbCredentialTypeUser);
795  if (!cred)
796  {
797  rfbClientLog("Reading credential failed\n");
798  break;
799  }
800

802  mod = (uint8_t*)malloc(keylen*4);
803  if (!mod)
804  {
805  rfbClientLog("malloc out of memory\n");
806  break;
807  }
808  resp = mod+keylen;
809  pub = resp+keylen;
810  key = pub+keylen;
811 
812  if (!ReadFromRFBServer(client, (char *)mod, keylen))
813  break;
814  if (!ReadFromRFBServer(client, (char *)resp, keylen))
815  break;
816 
817  error = gcry_mpi_scan(&genmpi, GCRYMPI_FMT_USG, gen, 2, NULL);
818  if (gcry_err_code(error) != GPG_ERR_NO_ERROR)
819  {
820  rfbClientLog("gcry_mpi_scan error: %s\n", gcry_strerror(error));
821  break;
822  }
823  error = gcry_mpi_scan(&modmpi, GCRYMPI_FMT_USG, mod, keylen, NULL);
824  if (gcry_err_code(error) != GPG_ERR_NO_ERROR)
825  {
826  rfbClientLog("gcry_mpi_scan error: %s\n", gcry_strerror(error));
827  break;
828  }
829  error = gcry_mpi_scan(&respmpi, GCRYMPI_FMT_USG, resp, keylen, NULL);
830  if (gcry_err_code(error) != GPG_ERR_NO_ERROR)
831  {
832  rfbClientLog("gcry_mpi_scan error: %s\n", gcry_strerror(error));
833  break;
834  }
835 
836  privmpi = gcry_mpi_new(keylen);
837  if (!privmpi)
838  {
839  rfbClientLog("gcry_mpi_new out of memory\n");
840  break;
841  }
842  gcry_mpi_randomize(privmpi, (keylen/8)*8, GCRY_STRONG_RANDOM);
843 
844  pubmpi = gcry_mpi_new(keylen);
845  if (!pubmpi)
846  {
847  rfbClientLog("gcry_mpi_new out of memory\n");
848  break;
849  }
850  gcry_mpi_powm(pubmpi, genmpi, privmpi, modmpi);
851 
852  keympi = gcry_mpi_new(keylen);
853  if (!keympi)
854  {
855  rfbClientLog("gcry_mpi_new out of memory\n");
856  break;
857  }
858  gcry_mpi_powm(keympi, respmpi, privmpi, modmpi);
859 
860  if (!rfbMpiToBytes(pubmpi, pub, keylen))
861  break;
862  if (!rfbMpiToBytes(keympi, key, keylen))
863  break;
864 
865  error = gcry_md_open(&md5, GCRY_MD_MD5, 0);
866  if (gcry_err_code(error) != GPG_ERR_NO_ERROR)
867  {
868  rfbClientLog("gcry_md_open error: %s\n", gcry_strerror(error));
869  break;
870  }
871  gcry_md_write(md5, key, keylen);
872  error = gcry_md_final(md5);
873  if (gcry_err_code(error) != GPG_ERR_NO_ERROR)
874  {
875  rfbClientLog("gcry_md_final error: %s\n", gcry_strerror(error));
876  break;
877  }
878  shared = gcry_md_read(md5, GCRY_MD_MD5);
879 
880  passwordLen = strlen(cred->userCredential.password)+1;
881  usernameLen = strlen(cred->userCredential.username)+1;
882  if (passwordLen > sizeof(userpass)/2)
883  passwordLen = sizeof(userpass)/2;
884  if (usernameLen > sizeof(userpass)/2)
885  usernameLen = sizeof(userpass)/2;
886 
887  gcry_randomize(userpass, sizeof(userpass), GCRY_STRONG_RANDOM);
888  memcpy(userpass, cred->userCredential.username, usernameLen);
889  memcpy(userpass+sizeof(userpass)/2, cred->userCredential.password, passwordLen);
890 
891  error = gcry_cipher_open(&aes, GCRY_CIPHER_AES128, GCRY_CIPHER_MODE_ECB, 0);
892  if (gcry_err_code(error) != GPG_ERR_NO_ERROR)
893  {
894  rfbClientLog("gcry_cipher_open error: %s\n", gcry_strerror(error));
895  break;
896  }
897  error = gcry_cipher_setkey(aes, shared, 16);
898  if (gcry_err_code(error) != GPG_ERR_NO_ERROR)
899  {
900  rfbClientLog("gcry_cipher_setkey error: %s\n", gcry_strerror(error));
901  break;
902  }
903  error = gcry_cipher_encrypt(aes, ciphertext, sizeof(ciphertext), userpass, sizeof(userpass));
904  if (gcry_err_code(error) != GPG_ERR_NO_ERROR)
905  {
906  rfbClientLog("gcry_cipher_encrypt error: %s\n", gcry_strerror(error));
907  break;
908  }
909 
910  if (!WriteToRFBServer(client, (char *)ciphertext, sizeof(ciphertext)))
911  break;
912  if (!WriteToRFBServer(client, (char *)pub, keylen))
913  break;
914 
915  /* Handle the SecurityResult message */
916  if (!rfbHandleAuthResult(client))
917  break;
918 
919  result = TRUE;
920  break;
921  }
922 
923  if (cred)
924  FreeUserCredential(cred);
925  if (mod)
926  free(mod);
927  if (genmpi)
928  gcry_mpi_release(genmpi);
929  if (modmpi)
930  gcry_mpi_release(modmpi);
931  if (respmpi)
932  gcry_mpi_release(respmpi);
933  if (privmpi)
934  gcry_mpi_release(privmpi);
935  if (pubmpi)
936  gcry_mpi_release(pubmpi);
937  if (keympi)
938  gcry_mpi_release(keympi);
939  if (md5)
940  gcry_md_close(md5);
941  if (aes)
942  gcry_cipher_close(aes);
943  return result;
944 }
