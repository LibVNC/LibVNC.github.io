87 struct rfbssl_ctx *rfbssl_init_global(char *key, char *cert)
88 {
89  int ret = GNUTLS_E_SUCCESS;
90  struct rfbssl_ctx *ctx = NULL;
91 
92  if (NULL == (ctx = malloc(sizeof(struct rfbssl_ctx)))) {
93  ret = GNUTLS_E_MEMORY_ERROR;
94  } else if (!GNUTLS_E_SUCCESS == (ret = gnutls_global_init())) {
95  /* */
96  } else if (!GNUTLS_E_SUCCESS == (ret = gnutls_certificate_allocate_credentials(&ctx->x509_cred))) {
97  /* */
98  } else if ((ret = gnutls_certificate_set_x509_trust_file(ctx->x509_cred, cert, GNUTLS_X509_FMT_PEM)) < 0) {
99  /* */
100  } else if (!GNUTLS_E_SUCCESS == (ret = gnutls_certificate_set_x509_key_file(ctx->x509_cred, cert, key, GNUTLS_X509_FMT_PEM))) {
101  /* */
102  } else if (!GNUTLS_E_SUCCESS == (ret = generate_dh_params(ctx))) {
103  /* */
104 #ifdef I_LIKE_RSA_PARAMS_THAT_MUCH
105  } else if (!GNUTLS_E_SUCCESS == (ret = generate_rsa_params(ctx))) {
106  /* */
107 #endif
108  } else {
109  gnutls_global_set_log_function(rfbssl_log_func);
110  gnutls_global_set_log_level(1);
111  gnutls_certificate_set_dh_params(ctx->x509_cred, ctx->dh_params);
112  /* newly allocated memory should be initialized, at least where it is important */
113  ctx->peekstart = ctx->peeklen = 0;
114  return ctx;
115  }
116 
117  free(ctx);
118  return NULL;
119 }

121 int rfbssl_init(rfbClientPtr cl)
122 {
123  int ret = -1;
124  struct rfbssl_ctx *ctx;
125  char *keyfile;
126  if (!(keyfile = cl->screen->sslkeyfile))
127  keyfile = cl->screen->sslcertfile;
128 
129  if (NULL == (ctx = rfbssl_init_global(keyfile, cl->screen->sslcertfile))) {
130  /* */
131  } else if (GNUTLS_E_SUCCESS != (ret = rfbssl_init_session(ctx, cl->sock))) {
132  /* */
133  } else {
134  while (GNUTLS_E_SUCCESS != (ret = gnutls_handshake(ctx->session))) {
135  if (ret == GNUTLS_E_AGAIN)
136  continue;
137  break;
138  }
139  }
140 
141  if (ret != GNUTLS_E_SUCCESS) {
142  rfbssl_error(__func__, ret);
143  } else {
144  cl->sslctx = (rfbSslCtx *)ctx;
145  rfbLog("%s protocol initialized\n", gnutls_protocol_get_name(gnutls_protocol_get_version(ctx->session)));
146  }
147  return ret;
148 }

150 static int rfbssl_do_read(rfbClientPtr cl, char *buf, int bufsize)
151 {
152  struct rfbssl_ctx *ctx = (struct rfbssl_ctx *)cl->sslctx;
153  int ret;
154 
155  while ((ret = gnutls_record_recv(ctx->session, buf, bufsize)) < 0) {
156  if (ret == GNUTLS_E_AGAIN) {
157  /* continue */
158  } else if (ret == GNUTLS_E_INTERRUPTED) {
159  /* continue */
160  } else {
161  break;
162  }
163  }
164 
165  if (ret < 0) {
166  rfbssl_error(__func__, ret);
167  errno = EIO;
168  ret = -1;
169  }
170 
171  return ret < 0 ? -1 : ret;
172 }

195 static void rfbssl_gc_peekbuf(struct rfbssl_ctx *ctx, int bufsize)
196 {
197  if (ctx->peekstart) {
198  int spaceleft = sizeof(ctx->peekbuf) - ctx->peeklen - ctx->peekstart;
199  if (spaceleft < bufsize) {

201  ctx->peekstart = 0;
202  }

206 static int __rfbssl_read(rfbClientPtr cl, char *buf, int bufsize, int peek)
207 {
208  int ret = 0;
209  struct rfbssl_ctx *ctx = (struct rfbssl_ctx *)cl->sslctx;
210 
211  rfbssl_gc_peekbuf(ctx, bufsize);
212 
213  if (ctx->peeklen) {
214  /* If we have any peek data, simply return that. */
215  ret = bufsize < ctx->peeklen ? bufsize : ctx->peeklen;
216  memcpy (buf, ctx->peekbuf + ctx->peekstart, ret);
217  if (!peek) {
218  ctx->peeklen -= ret;
219  if (ctx->peeklen != 0)
220  ctx->peekstart += ret;
221  else
222  ctx->peekstart = 0;
223  }
224  }
225 
226  if (ret < bufsize) {
227  int n;
228  /* read the remaining data */
229  if ((n = rfbssl_do_read(cl, buf + ret, bufsize - ret)) <= 0) {
230  rfbErr("rfbssl_%s: %s error\n", __func__, peek ? "peek" : "read");
231  return n;
232  }
233  if (peek) {
234  memcpy(ctx->peekbuf + ctx->peekstart + ctx->peeklen, buf + ret, n);
235  ctx->peeklen += n;
236  }
237  ret += n;
238  }
239 
240  return ret;
241 }
