191  * encode current video frame and send it to the muxer

194 int write_video_frame(AVFormatContext *oc, VideoOutputStream *ost, int64_t pts)
195 {
196  int ret, ret2;
197  AVPacket pkt = { 0 };
198  if (pts <= ost->pts) return 0; // nothing to do
199  /* convert format if needed */
200  if (ost->tmp_frame) {
201  sws_scale(ost->sws, (const uint8_t * const *)ost->tmp_frame->data,
202  ost->tmp_frame->linesize, 0, ost->enc->height, ost->frame->data, ost->frame->linesize);
203  }
204 
205  /* send the imager to encoder */
206  ost->pts = pts;
207  ost->frame->pts = ost->pts;
208  ret = avcodec_send_frame(ost->enc, ost->frame);
209  if (ret < 0) {
210  fprintf(stderr, "Error sending video frame to encoder: %s\n", av_err2str(ret));
211  return ret;
212  }
213  /* read all available packets */
214  ret2 = 0;
215  for (ret = avcodec_receive_packet(ost->enc, &pkt); ret == 0; ret = avcodec_receive_packet(ost->enc, &pkt)) {
216  ret2 = write_packet(oc, &(ost->enc->time_base), ost->st, &pkt);
217  if (ret2 < 0) {
218  fprintf(stderr, "Error while writing video frame: %s\n", av_err2str(ret2));
219  /* continue on this error to not gum up encoder */
220  }
221  }
222  if (ret2 < 0) return ret2;
223  if (!(ret == AVERROR(EAGAIN))) return ret; // if AVERROR(EAGAIN), means all available packets output, need more frames (i.e. success)
224  return 0;
225 }

371 /* returns time since start in pts units */
372 int64_t time_to_pts(int framerate, struct timespec *start_time, struct timespec *cur_time) {
373  time_t ds = cur_time->tv_sec - start_time->tv_sec;
374  long dns = cur_time->tv_nsec - start_time->tv_nsec;
375  /* use usecs */
376  int64_t dt = (int64_t)ds*(int64_t)1000000+(int64_t)dns/(int64_t)1000;
377  /* compute rv in units of frame number (rounding to nearest, not truncating) */
378  int64_t rv = (((int64_t)framerate)*dt + (int64_t)500000) / (int64_t)(1000000);
379 
380  return rv;
381 }
