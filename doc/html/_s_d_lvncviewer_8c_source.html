186 /* UTF-8 decoding is from https://rosettacode.org/wiki/UTF-8_encode_and_decode which is under GFDL 1.2 */
187 static rfbKeySym utf8char2rfbKeySym(const char chr[4]) {
188  int bytes = strlen(chr);
189  int shift = utf8Mapping[0].bits_stored * (bytes - 1);
190  rfbKeySym codep = (*chr++ & utf8Mapping[bytes].mask) << shift;
191  int i;
192  for(i = 1; i < bytes; ++i, ++chr) {
193  shift -= utf8Mapping[0].bits_stored;
194  codep |= ((char)*chr & utf8Mapping[0].mask) << shift;
195  }
196  return codep;
197 }

293 static rfbBool handleSDLEvent(rfbClient *cl, SDL_Event *e)
294 {
295  switch(e->type) {
296  case SDL_WINDOWEVENT:
297  switch (e->window.event) {
298  case SDL_WINDOWEVENT_EXPOSED:
299  SendFramebufferUpdateRequest(cl, 0, 0,
300  cl->width, cl->height, FALSE);
301  break;
302  case SDL_WINDOWEVENT_FOCUS_GAINED:
303  if (SDL_HasClipboardText()) {
304  char *text = SDL_GetClipboardText();
305  if(text) {
306  rfbClientLog("sending clipboard text '%s'\n", text);
307  SendClientCutText(cl, text, strlen(text));
308  }
309  }
310 
311  break;
312  case SDL_WINDOWEVENT_FOCUS_LOST:
313  if (rightAltKeyDown) {
314  SendKeyEvent(cl, XK_Alt_R, FALSE);
315  rightAltKeyDown = FALSE;
316  rfbClientLog("released right Alt key\n");
317  }
318  if (leftAltKeyDown) {
319  SendKeyEvent(cl, XK_Alt_L, FALSE);
320  leftAltKeyDown = FALSE;
321  rfbClientLog("released left Alt key\n");
322  }
323  break;
324  }
325  break;
326  case SDL_MOUSEWHEEL:
327  {
328  int steps;
329  if (viewOnly)
330  break;
331 
332  if(e->wheel.y > 0)
333  for(steps = 0; steps < e->wheel.y; ++steps) {
334  SendPointerEvent(cl, x, y, rfbButton4Mask);
335  SendPointerEvent(cl, x, y, 0);
336  }
337  if(e->wheel.y < 0)
338  for(steps = 0; steps > e->wheel.y; --steps) {
339  SendPointerEvent(cl, x, y, rfbButton5Mask);
340  SendPointerEvent(cl, x, y, 0);
341  }
342  if(e->wheel.x > 0)
343  for(steps = 0; steps < e->wheel.x; ++steps) {
344  SendPointerEvent(cl, x, y, 0b01000000);
345  SendPointerEvent(cl, x, y, 0);
346  }
347  if(e->wheel.x < 0)
348  for(steps = 0; steps > e->wheel.x; --steps) {
349  SendPointerEvent(cl, x, y, 0b00100000);
350  SendPointerEvent(cl, x, y, 0);
351  }
352  break;
353  }
354  case SDL_MOUSEBUTTONUP:
355  case SDL_MOUSEBUTTONDOWN:
356  case SDL_MOUSEMOTION:
357  {
358  int state, i;
359  if (viewOnly)
360  break;
361 
362  if (e->type == SDL_MOUSEMOTION) {
363  x = e->motion.x;
364  y = e->motion.y;
365  state = e->motion.state;
366  }
367  else {
368  x = e->button.x;
369  y = e->button.y;
370  state = e->button.button;
371  for (i = 0; buttonMapping[i].sdl; i++)
372  if (state == buttonMapping[i].sdl) {
373  state = buttonMapping[i].rfb;
374  if (e->type == SDL_MOUSEBUTTONDOWN)
375  buttonMask |= state;
376  else
377  buttonMask &= ~state;
378  break;
379  }
380  }
381  SendPointerEvent(cl, x, y, buttonMask);
382  buttonMask &= ~(rfbButton4Mask | rfbButton5Mask);
383  break;
384  }
385  case SDL_KEYUP:
386  case SDL_KEYDOWN:
387  if (viewOnly)
388  break;
389  SendKeyEvent(cl, SDL_key2rfbKeySym(&e->key),
390  e->type == SDL_KEYDOWN ? TRUE : FALSE);
391  if (e->key.keysym.sym == SDLK_RALT)
392  rightAltKeyDown = e->type == SDL_KEYDOWN;
393  if (e->key.keysym.sym == SDLK_LALT)
394  leftAltKeyDown = e->type == SDL_KEYDOWN;
395  break;
396  case SDL_TEXTINPUT:
397  if (viewOnly)
398  break;
399  rfbKeySym sym = utf8char2rfbKeySym(e->text.text);
400  SendKeyEvent(cl, sym, TRUE);
401  SendKeyEvent(cl, sym, FALSE);
402  break;
403  case SDL_QUIT:
404  if(listenLoop)
405  {
406  cleanup(cl);
407  return FALSE;
408  }
409  else
410  {
411  rfbClientCleanup(cl);
412  exit(0);
413  }
414  default:
415  rfbClientLog("ignore SDL event: 0x%x\n", e->type);
416  }
417  return TRUE;
418 }
