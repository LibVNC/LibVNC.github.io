38static rfbBool resize(rfbClient* client) {
39 int width=client->width,height=client->height,
40 depth=client->format.bitsPerPixel;
41
42 if (enableResizable)
43 sdlFlags |= SDL_WINDOW_RESIZABLE;
44
45 /* (re)create the surface used as the client's framebuffer */
46 SDL_FreeSurface(rfbClientGetClientData(client, SDL_Init));
47 SDL_Surface* sdl=SDL_CreateRGBSurface(0,
48 width,
49 height,
50 depth,
51 0,0,0,0);
52 if(!sdl)
53 rfbClientErr("resize: error creating surface: %s\n", SDL_GetError());
54
55 rfbClientSetClientData(client, SDL_Init, sdl);
56 client->width = sdl->pitch / (depth / 8);
57 client->frameBuffer=sdl->pixels;
58
59 client->format.bitsPerPixel=depth;
60 client->format.redShift=sdl->format->Rshift;
61 client->format.greenShift=sdl->format->Gshift;
62 client->format.blueShift=sdl->format->Bshift;
63 client->format.redMax=sdl->format->Rmask>>client->format.redShift;
64 client->format.greenMax=sdl->format->Gmask>>client->format.greenShift;
65 client->format.blueMax=sdl->format->Bmask>>client->format.blueShift;
66 SetFormatAndEncodings(client);
67
68 /* create or resize the window */
69 if(!sdlWindow) {
70 sdlWindow = SDL_CreateWindow(client->desktopName,
71 SDL_WINDOWPOS_UNDEFINED,
72 SDL_WINDOWPOS_UNDEFINED,
73 width,
74 height,
75 sdlFlags);
76 if(!sdlWindow)
77 rfbClientErr("resize: error creating window: %s\n", SDL_GetError());
78 } else {
79 SDL_SetWindowSize(sdlWindow, width, height);
80 }
81
82 /* create the renderer if it does not already exist */
83 if(!sdlRenderer) {
84 sdlRenderer = SDL_CreateRenderer(sdlWindow, -1, 0);
85 if(!sdlRenderer)
86 rfbClientErr("resize: error creating renderer: %s\n", SDL_GetError());
87 SDL_SetHint(SDL_HINT_RENDER_SCALE_QUALITY, "linear"); /* make the scaled rendering look smoother. */
88 }
89 SDL_RenderSetLogicalSize(sdlRenderer, width, height); /* this is a departure from the SDL1.2-based version, but more in the sense of a VNC viewer in keeeping aspect ratio */
90
91 /* (re)create the texture that sits in between the surface->pixels and the renderer */
92 if(sdlTexture)
93 SDL_DestroyTexture(sdlTexture);
94 sdlTexture = SDL_CreateTexture(sdlRenderer,
95 SDL_PIXELFORMAT_ARGB8888,
96 SDL_TEXTUREACCESS_STREAMING,
97 width, height);
98 if(!sdlTexture)
99 rfbClientErr("resize: error creating texture: %s\n", SDL_GetError());
100 return TRUE;
101}

183/* UTF-8 decoding is from https://rosettacode.org/wiki/UTF-8_encode_and_decode which is under GFDL 1.2 */
184static rfbKeySym utf8char2rfbKeySym(const char chr[4]) {
185 int bytes = strlen(chr);
186 int shift = utf8Mapping[0].bits_stored * (bytes - 1);
187 rfbKeySym codep = (*chr++ & utf8Mapping[bytes].mask) << shift;
188 int i;
189 for(i = 1; i < bytes; ++i, ++chr) {
190 shift -= utf8Mapping[0].bits_stored;
191 codep |= ((char)*chr & utf8Mapping[0].mask) << shift;
192 }
193 return codep;
194}

196static void update(rfbClient* cl,int x,int y,int w,int h) {
197 SDL_Surface *sdl = rfbClientGetClientData(cl, SDL_Init);
198 /* update texture from surface->pixels */
199 SDL_Rect r = {x,y,w,h};

201 rfbClientErr("update: failed to update texture: %s\n", SDL_GetError());

290static rfbBool handleSDLEvent(rfbClient *cl, SDL_Event *e)
291{
292 switch(e->type) {
293 case SDL_WINDOWEVENT:
294 switch (e->window.event) {
295 case SDL_WINDOWEVENT_EXPOSED:
296 SendFramebufferUpdateRequest(cl, 0, 0,
297 cl->width, cl->height, FALSE);
298 break;
299 case SDL_WINDOWEVENT_RESIZED:
300 SendExtDesktopSize(cl, e->window.data1, e->window.data2);
301 break;
302 case SDL_WINDOWEVENT_FOCUS_GAINED:
303 if (SDL_HasClipboardText()) {
304 char *text = SDL_GetClipboardText();
305 if(text) {
306 rfbClientLog("sending clipboard text '%s'\n", text);
307 if(!SendClientCutTextUTF8(cl, text, strlen(text)))
308 SendClientCutText(cl, text, strlen(text));
309 }
310 }
311
312 break;
313 case SDL_WINDOWEVENT_FOCUS_LOST:
314 if (rightAltKeyDown) {
315 SendKeyEvent(cl, XK_Alt_R, FALSE);
316 rightAltKeyDown = FALSE;
317 rfbClientLog("released right Alt key\n");
318 }
319 if (leftAltKeyDown) {
320 SendKeyEvent(cl, XK_Alt_L, FALSE);
321 leftAltKeyDown = FALSE;
322 rfbClientLog("released left Alt key\n");
323 }
324 break;
325 }
326 break;
327 case SDL_MOUSEWHEEL:
328 {
329 int steps;
330 if (viewOnly)
331 break;
332
333 if(e->wheel.y > 0)
334 for(steps = 0; steps < e->wheel.y; ++steps) {
335 SendPointerEvent(cl, x, y, rfbButton4Mask);
336 SendPointerEvent(cl, x, y, 0);
337 }
338 if(e->wheel.y < 0)
339 for(steps = 0; steps > e->wheel.y; --steps) {
340 SendPointerEvent(cl, x, y, rfbButton5Mask);
341 SendPointerEvent(cl, x, y, 0);
342 }
343 if(e->wheel.x > 0)
344 for(steps = 0; steps < e->wheel.x; ++steps) {
345 SendPointerEvent(cl, x, y, 0b01000000);
346 SendPointerEvent(cl, x, y, 0);
347 }
348 if(e->wheel.x < 0)
349 for(steps = 0; steps > e->wheel.x; --steps) {
350 SendPointerEvent(cl, x, y, 0b00100000);
351 SendPointerEvent(cl, x, y, 0);
352 }
353 break;
354 }
355 case SDL_MOUSEBUTTONUP:
356 case SDL_MOUSEBUTTONDOWN:
357 case SDL_MOUSEMOTION:
358 {
359 int state, i;
360 if (viewOnly)
361 break;
362
363 if (e->type == SDL_MOUSEMOTION) {
364 x = e->motion.x;
365 y = e->motion.y;
366 state = e->motion.state;
367 }
368 else {
369 x = e->button.x;
370 y = e->button.y;
371 state = e->button.button;
372 for (i = 0; buttonMapping[i].sdl; i++)
373 if (state == buttonMapping[i].sdl) {
374 state = buttonMapping[i].rfb;
375 if (e->type == SDL_MOUSEBUTTONDOWN)
376 buttonMask |= state;
377 else
378 buttonMask &= ~state;
379 break;
380 }
381 }
382 SendPointerEvent(cl, x, y, buttonMask);
383 buttonMask &= ~(rfbButton4Mask | rfbButton5Mask);
384 break;
385 }
386 case SDL_KEYUP:
387 case SDL_KEYDOWN:
388 if (viewOnly)
389 break;
390 SendKeyEvent(cl, SDL_key2rfbKeySym(&e->key),
391 e->type == SDL_KEYDOWN ? TRUE : FALSE);
392 if (e->key.keysym.sym == SDLK_RALT)
393 rightAltKeyDown = e->type == SDL_KEYDOWN;
394 if (e->key.keysym.sym == SDLK_LALT)
395 leftAltKeyDown = e->type == SDL_KEYDOWN;
396 break;
397 case SDL_TEXTINPUT:
398 if (viewOnly)
399 break;
400 rfbKeySym sym = utf8char2rfbKeySym(e->text.text);
401 SendKeyEvent(cl, sym, TRUE);
402 SendKeyEvent(cl, sym, FALSE);
403 break;
404 case SDL_QUIT:
405 if(listenLoop)
406 {
407 cleanup(cl);
408 return FALSE;
409 }
410 else
411 {
412 rfbClientCleanup(cl);
413 exit(0);
414 }
415 default:
416 rfbClientLog("ignore SDL event: 0x%x\n", e->type);
417 }
418 return TRUE;
419}

436static rfbCredential* get_credential(rfbClient* cl, int credentialType){
437 rfbCredential *c = malloc(sizeof(rfbCredential));
438 if (!c) {
439 return NULL;
440 }
441 c->userCredential.username = malloc(RFB_BUF_SIZE);
442 if (!c->userCredential.username) {
443 free(c);
444 return NULL;
445 }
446 c->userCredential.password = malloc(RFB_BUF_SIZE);
447 if (!c->userCredential.password) {
448 free(c->userCredential.username);
449 free(c);
450 return NULL;
451 }
452
453 if(credentialType != rfbCredentialTypeUser) {
454 rfbClientErr("something else than username and password required for authentication\n");
455 return NULL;
456 }
457
458 rfbClientLog("username and password required for authentication!\n");
459 printf("user: ");
460 fgets(c->userCredential.username, RFB_BUF_SIZE, stdin);
461 printf("pass: ");
462 fgets(c->userCredential.password, RFB_BUF_SIZE, stdin);
463
464 /* remove trailing newlines */
465 c->userCredential.username[strcspn(c->userCredential.username, "\n")] = 0;
466 c->userCredential.password[strcspn(c->userCredential.password, "\n")] = 0;
467
468 return c;
469}

476int main(int argc,char** argv) {
477 rfbClient* cl;
478 int i, j;
479 SDL_Event e;
480
481#ifdef LOG_TO_FILE
482 rfbClientLog=rfbClientErr=log_to_file;
483#endif
484
485 for (i = 1, j = 1; i < argc; i++)
486 if (!strcmp(argv[i], "-viewonly"))
487 viewOnly = 1;
488 else if (!strcmp(argv[i], "-resizable"))
489 enableResizable = 1;
490 else if (!strcmp(argv[i], "-no-resizable"))
491 enableResizable = 0;
492 else if (!strcmp(argv[i], "-listen")) {
493 listenLoop = 1;
494 argv[i] = "-listennofork";
495 ++j;
496 }
497 else {
498 if (i != j)
499 argv[j] = argv[i];
500 j++;
501 }
502 argc = j;
503
504 SDL_Init(SDL_INIT_VIDEO | SDL_INIT_NOPARACHUTE);
505 atexit(SDL_Quit);
506 signal(SIGINT, exit);
507
508 do {
509 /* 16-bit: cl=rfbGetClient(5,3,2); */
510 cl=rfbGetClient(8,3,4);
511 cl->MallocFrameBuffer=resize;
512 cl->canHandleNewFBSize = TRUE;
513 cl->GotFrameBufferUpdate=update;
514 cl->HandleKeyboardLedState=kbd_leds;
515 cl->HandleTextChat=text_chat;
516 /* two different cut text handlers here for demo purposes, you
517 might as well use the same callback for both if it doesn't
518 matter for your application */
519 cl->GotXCutText = got_selection_latin1;
520 cl->GotXCutTextUTF8 = got_selection_utf8;
521 cl->GetCredential = get_credential;
522 cl->listenPort = LISTEN_PORT_OFFSET;
523 cl->listen6Port = LISTEN_PORT_OFFSET;
524 if(!rfbInitClient(cl,&argc,argv))
525 {
526 cl = NULL; /* rfbInitClient has already freed the client struct */
527 cleanup(cl);
528 break;
529 }
530
531 while(1) {
532 if(SDL_PollEvent(&e)) {
533 /*
534 handleSDLEvent() return 0 if user requested window close.
535 In this case, handleSDLEvent() will have called cleanup().
536 */
537 if(!handleSDLEvent(cl, &e))
538 break;
539 }
540 else {
541 i=WaitForMessage(cl,500);
542 if(i<0)
543 {
544 cleanup(cl);
545 break;
546 }
547 if(i)
548 if(!HandleRFBServerMessage(cl))
549 {
550 cleanup(cl);
551 break;
552 }
553 }
554 }
555 }
556 while(listenLoop);
557
558 return 0;
559}
