48 /*
49  * ReadFromRFBServer is called whenever we want to read some data from the RFB
50  * server. It is non-trivial for two reasons:
51  *
52  * 1. For efficiency it performs some intelligent buffering, avoiding invoking
53  * the read() system call too often. For small chunks of data, it simply
54  * copies the data out of an internal buffer. For large amounts of data it
55  * reads directly into the buffer provided by the caller.
56  *
57  * 2. Whenever read() would block, it invokes the Xt event dispatching
58  * mechanism to process X events. In fact, this is the only place these
59  * events are processed, as there is no XtAppMainLoop in the program.
60  */
61 
62 rfbBool
63 ReadFromRFBServer(rfbClient* client, char *out, unsigned int n)
64 {
65  const int USECS_WAIT_PER_RETRY = 100000;
66  int retries = 0;
67 #undef DEBUG_READ_EXACT
68 #ifdef DEBUG_READ_EXACT
69  char* oout=out;
70  unsigned int nn=n;
71  rfbClientLog("ReadFromRFBServer %d bytes\n",n);
72 #endif
73 
74  /* Handle attempts to write to NULL out buffer that might occur
75  when an outside malloc() fails. For instance, memcpy() to NULL
76  results in undefined behaviour and probably memory corruption.*/
77  if(!out)
78  return FALSE;
79 
80  if (client->serverPort==-1) {
81  /* vncrec playing */
82  rfbVNCRec* rec = client->vncRec;
83  struct timeval tv;
84 
85  if (rec->readTimestamp) {
86  rec->readTimestamp = FALSE;
87  if (!fread(&tv,sizeof(struct timeval),1,rec->file))
88  return FALSE;
89 
90  tv.tv_sec = rfbClientSwap32IfLE (tv.tv_sec);
91  tv.tv_usec = rfbClientSwap32IfLE (tv.tv_usec);
92 
93  if (rec->tv.tv_sec!=0 && !rec->doNotSleep) {
94  struct timeval diff;
95  diff.tv_sec = tv.tv_sec - rec->tv.tv_sec;
96  diff.tv_usec = tv.tv_usec - rec->tv.tv_usec;
97  if(diff.tv_usec<0) {
98  diff.tv_sec--;
99  diff.tv_usec+=1000000;
100  }
101 #ifndef WIN32
102  sleep (diff.tv_sec);
103  usleep (diff.tv_usec);
104 #else
105  Sleep (diff.tv_sec * 1000 + diff.tv_usec/1000);
106 #endif
107  }
108 
109  rec->tv=tv;
110  }
111 
112  return (fread(out,1,n,rec->file) != n ? FALSE : TRUE);
113  }
114 
115  if (n <= client->buffered) {
116  memcpy(out, client->bufoutptr, n);
117  client->bufoutptr += n;
118  client->buffered -= n;
119 #ifdef DEBUG_READ_EXACT
120  goto hexdump;
121 #endif
122  return TRUE;
123  }
124 
125  memcpy(out, client->bufoutptr, client->buffered);
126 
127  out += client->buffered;
128  n -= client->buffered;
129 
130  client->bufoutptr = client->buf;
131  client->buffered = 0;
132 
133  if (n <= RFB_BUF_SIZE) {
134 
135  while (client->buffered < n) {
136  int i;
137  if (client->tlsSession)
138  i = ReadFromTLS(client, client->buf + client->buffered, RFB_BUF_SIZE - client->buffered);
139  else
140 #ifdef LIBVNCSERVER_HAVE_SASL
141  if (client->saslconn)
142  i = ReadFromSASL(client, client->buf + client->buffered, RFB_BUF_SIZE - client->buffered);
143  else {
144 #endif /* LIBVNCSERVER_HAVE_SASL */
145  i = read(client->sock, client->buf + client->buffered, RFB_BUF_SIZE - client->buffered);
146 #ifdef WIN32
147  if (i < 0) errno=WSAGetLastError();
148 #endif
149 #ifdef LIBVNCSERVER_HAVE_SASL
150  }
151 #endif
152 
153  if (i <= 0) {
154  if (i < 0) {
155  if (errno == EWOULDBLOCK || errno == EAGAIN) {
156  if (client->readTimeout > 0 &&
157  ++retries > (client->readTimeout * 1000 * 1000 / USECS_WAIT_PER_RETRY))
158  {
159  rfbClientLog("Connection timed out\n");
160  return FALSE;
161  }
162  /* TODO:
163  ProcessXtEvents();
164  */
165  WaitForMessage(client, USECS_WAIT_PER_RETRY);
166  i = 0;
167  } else {
168  rfbClientErr("read (%d: %s)\n",errno,strerror(errno));
169  return FALSE;
170  }
171  } else {
172  if (errorMessageOnReadFailure) {
173  rfbClientLog("VNC server closed connection\n");
174  }
175  return FALSE;
176  }
177  }
178  client->buffered += i;
179  }
180 
181  memcpy(out, client->bufoutptr, n);
182  client->bufoutptr += n;
183  client->buffered -= n;
184 
185  } else {
186 
187  while (n > 0) {
188  int i;
189  if (client->tlsSession)
190  i = ReadFromTLS(client, out, n);
191  else
192 #ifdef LIBVNCSERVER_HAVE_SASL
193  if (client->saslconn)
194  i = ReadFromSASL(client, out, n);
195  else
196 #endif
197  i = read(client->sock, out, n);
198 
199  if (i <= 0) {
200  if (i < 0) {
201 #ifdef WIN32
202  errno=WSAGetLastError();
203 #endif
204  if (errno == EWOULDBLOCK || errno == EAGAIN) {
205  if (client->readTimeout > 0 &&
206  ++retries > (client->readTimeout * 1000 * 1000 / USECS_WAIT_PER_RETRY))
207  {
208  rfbClientLog("Connection timed out\n");
209  return FALSE;
210  }
211  /* TODO:
212  ProcessXtEvents();
213  */
214  WaitForMessage(client, USECS_WAIT_PER_RETRY);
215  i = 0;
216  } else {
217  rfbClientErr("read (%s)\n",strerror(errno));
218  return FALSE;
219  }
220  } else {
221  if (errorMessageOnReadFailure) {
222  rfbClientLog("VNC server closed connection\n");
223  }
224  return FALSE;
225  }
226  }
227  out += i;
228  n -= i;
229  }
230  }
231 
232 #ifdef DEBUG_READ_EXACT
233 hexdump:
234  { unsigned int ii;
235  for(ii=0;ii<nn;ii++)
236  fprintf(stderr,"%02x ",(unsigned char)oout[ii]);
237  fprintf(stderr,"\n");
238  }
239 #endif
240 
241  return TRUE;
242 }

249 rfbBool
250 WriteToRFBServer(rfbClient* client, const char *buf, unsigned int n)
251 {
252  fd_set fds;
253  int i = 0;
254  int j;
255  const char *obuf = buf;
256 #ifdef LIBVNCSERVER_HAVE_SASL
257  const char *output;
258  unsigned int outputlen;
259  int err;
260 #endif /* LIBVNCSERVER_HAVE_SASL */
261 
262  if (client->serverPort==-1)
263  return TRUE; /* vncrec playing */
264 
265  if (client->tlsSession) {
266  /* WriteToTLS() will guarantee either everything is written, or error/eof returns */
267  i = WriteToTLS(client, buf, n);
268  if (i <= 0) return FALSE;
269 
270  return TRUE;
271  }
272 #ifdef LIBVNCSERVER_HAVE_SASL
273  if (client->saslconn) {
274  err = sasl_encode(client->saslconn,
275  buf, n,
276  &output, &outputlen);
277  if (err != SASL_OK) {
278  rfbClientLog("Failed to encode SASL data %s",
279  sasl_errstring(err, NULL, NULL));
280  return FALSE;
281  }
282  obuf = output;
283  n = outputlen;
284  }
285 #endif /* LIBVNCSERVER_HAVE_SASL */
286 
287  while (i < n) {
288  j = write(client->sock, obuf + i, (n - i));
289  if (j <= 0) {
290  if (j < 0) {
291 #ifdef WIN32
292  errno=WSAGetLastError();
293 #endif
294  if (errno == EWOULDBLOCK ||
295 #ifdef LIBVNCSERVER_ENOENT_WORKAROUND
296  errno == ENOENT ||
297 #endif
298  errno == EAGAIN) {
299  FD_ZERO(&fds);
300  FD_SET(client->sock,&fds);
301 
302  if (select(client->sock+1, NULL, &fds, NULL, NULL) <= 0) {
303  rfbClientErr("select\n");
304  return FALSE;
305  }
306  j = 0;
307  } else {
308  rfbClientErr("write\n");
309  return FALSE;
310  }
311  } else {
312  rfbClientLog("write failed\n");
313  return FALSE;
314  }
315  }
316  i += j;
317  }
318  return TRUE;
319 }

322 static rfbBool WaitForConnected(int socket, unsigned int secs)
323 {
324  fd_set writefds;
325  fd_set exceptfds;
326  struct timeval timeout;
327 
328  timeout.tv_sec=secs;
329  timeout.tv_usec=0;
330 
331  FD_ZERO(&writefds);
332  FD_SET(socket, &writefds);
333  FD_ZERO(&exceptfds);
334  FD_SET(socket, &exceptfds);
335  if (select(socket+1, NULL, &writefds, &exceptfds, &timeout)==1) {
336 #ifdef WIN32
337  if (FD_ISSET(socket, &exceptfds))
338  return FALSE;
339 #else
340  int so_error;
341  socklen_t len = sizeof so_error;
342  getsockopt(socket, SOL_SOCKET, SO_ERROR, &so_error, &len);
343  if (so_error!=0)
344  return FALSE;
345 #endif
346  return TRUE;
347  }
348 
349  return FALSE;
350 }

365 ConnectClientToTcpAddrWithTimeout(unsigned int host, int port, unsigned int timeout)
366 {
367  rfbSocket sock;
368  struct sockaddr_in addr;
369  int one = 1;
370 
371  addr.sin_family = AF_INET;
372  addr.sin_port = htons(port);
373  addr.sin_addr.s_addr = host;
374 
375  sock = socket(AF_INET, SOCK_STREAM, 0);
376  if (sock == RFB_INVALID_SOCKET) {
377 #ifdef WIN32
378  errno=WSAGetLastError();
379 #endif
380  rfbClientErr("ConnectToTcpAddr: socket (%s)\n",strerror(errno));
381  return RFB_INVALID_SOCKET;
382  }
383 
384  if (!SetNonBlocking(sock))
385  return FALSE;
386 
387  if (connect(sock, (struct sockaddr *)&addr, sizeof(addr)) < 0) {
388 #ifdef WIN32
389  errno=WSAGetLastError();
390 #endif
391  if (!((errno == EWOULDBLOCK || errno == EINPROGRESS) && WaitForConnected(sock, timeout))) {
392  rfbClientErr("ConnectToTcpAddr: connect\n");
393  rfbCloseSocket(sock);
394  return RFB_INVALID_SOCKET;
395  }
396  }
397 
398  if (setsockopt(sock, IPPROTO_TCP, TCP_NODELAY,
399  (char *)&one, sizeof(one)) < 0) {
400  rfbClientErr("ConnectToTcpAddr: setsockopt\n");
401  rfbCloseSocket(sock);
402  return RFB_INVALID_SOCKET;
403  }
404 
405  return sock;

590 rfbSocket
591 ListenAtTcpPortAndAddress(int port, const char *address)
592 {
593  rfbSocket sock = RFB_INVALID_SOCKET;
594  int one = 1;
595 #ifndef LIBVNCSERVER_IPv6
596  struct sockaddr_in addr;
597 
598  addr.sin_family = AF_INET;
599  addr.sin_port = htons(port);
600  if (address) {
601  addr.sin_addr.s_addr = inet_addr(address);
602  } else {
603  addr.sin_addr.s_addr = htonl(INADDR_ANY);
604  }
605 
606  sock = socket(AF_INET, SOCK_STREAM, 0);
607  if (sock == RFB_INVALID_SOCKET) {
608  rfbClientErr("ListenAtTcpPort: socket\n");
609  return RFB_INVALID_SOCKET;
610  }
611 
612  if (setsockopt(sock, SOL_SOCKET, SO_REUSEADDR,
613  (const char *)&one, sizeof(one)) < 0) {
614  rfbClientErr("ListenAtTcpPort: setsockopt\n");
615  rfbCloseSocket(sock);
616  return RFB_INVALID_SOCKET;
617  }
618 
619  if (bind(sock, (struct sockaddr *)&addr, sizeof(addr)) < 0) {
620  rfbClientErr("ListenAtTcpPort: bind\n");
621  rfbCloseSocket(sock);
622  return RFB_INVALID_SOCKET;
623  }
624 
625 #else
626  int rv;
627  struct addrinfo hints, *servinfo, *p;
628  char port_str[8];
629 
630  snprintf(port_str, 8, "%d", port);
631 
632  memset(&hints, 0, sizeof(hints));
633  hints.ai_family = AF_UNSPEC;
634  hints.ai_socktype = SOCK_STREAM;
635  hints.ai_flags = AI_PASSIVE; /* fill in wildcard address if address == NULL */
636 
637  if ((rv = getaddrinfo(address, port_str, &hints, &servinfo)) != 0) {
638  rfbClientErr("ListenAtTcpPortAndAddress: error in getaddrinfo: %s\n", gai_strerror(rv));
639  return RFB_INVALID_SOCKET;
640  }
641 
642  /* loop through all the results and bind to the first we can */
643  for(p = servinfo; p != NULL; p = p->ai_next) {
644  if ((sock = socket(p->ai_family, p->ai_socktype, p->ai_protocol)) == RFB_INVALID_SOCKET) {
645  continue;
646  }
647 
648 #ifdef IPV6_V6ONLY
649  /* we have separate IPv4 and IPv6 sockets since some OS's do not support dual binding */
650  if (p->ai_family == AF_INET6 && setsockopt(sock, IPPROTO_IPV6, IPV6_V6ONLY, (char *)&one, sizeof(one)) < 0) {
651  rfbClientErr("ListenAtTcpPortAndAddress: error in setsockopt IPV6_V6ONLY: %s\n", strerror(errno));
652  rfbCloseSocket(sock);
653  freeaddrinfo(servinfo);
654  return RFB_INVALID_SOCKET;
655  }
656 #endif
657 
658  if (setsockopt(sock, SOL_SOCKET, SO_REUSEADDR, (char *)&one, sizeof(one)) < 0) {
659  rfbClientErr("ListenAtTcpPortAndAddress: error in setsockopt SO_REUSEADDR: %s\n", strerror(errno));
660  rfbCloseSocket(sock);
661  freeaddrinfo(servinfo);
662  return RFB_INVALID_SOCKET;
663  }
664 
665  if (bind(sock, p->ai_addr, p->ai_addrlen) < 0) {
666  rfbCloseSocket(sock);
667  continue;
668  }
669 
670  break;
671  }
672 
673  if (p == NULL) {
674  rfbClientErr("ListenAtTcpPortAndAddress: error in bind: %s\n", strerror(errno));
675  return RFB_INVALID_SOCKET;
676  }
677 
678  /* all done with this structure now */
679  freeaddrinfo(servinfo);
680 #endif
681 
682  if (listen(sock, 5) < 0) {
683  rfbClientErr("ListenAtTcpPort: listen\n");
684  rfbCloseSocket(sock);
685  return RFB_INVALID_SOCKET;
686  }
687 
688  return sock;
689 }

724 
725 rfbBool
726 SetNonBlocking(rfbSocket sock)
727 {
