71 /*
72  * ReadFromRFBServer is called whenever we want to read some data from the RFB
73  * server. It is non-trivial for two reasons:
74  *
75  * 1. For efficiency it performs some intelligent buffering, avoiding invoking
76  * the read() system call too often. For small chunks of data, it simply
77  * copies the data out of an internal buffer. For large amounts of data it
78  * reads directly into the buffer provided by the caller.
79  *
80  * 2. Whenever read() would block, it invokes the Xt event dispatching
81  * mechanism to process X events. In fact, this is the only place these
82  * events are processed, as there is no XtAppMainLoop in the program.
83  */
84 
85 rfbBool
86 ReadFromRFBServer(rfbClient* client, char *out, unsigned int n)
87 {
88 #undef DEBUG_READ_EXACT
89 #ifdef DEBUG_READ_EXACT
90  char* oout=out;
91  int nn=n;
92  rfbClientLog("ReadFromRFBServer %d bytes\n",n);
93 #endif
94 
95  /* Handle attempts to write to NULL out buffer that might occur
96  when an outside malloc() fails. For instance, memcpy() to NULL
97  results in undefined behaviour and probably memory corruption.*/
98  if(!out)
99  return FALSE;
100 
101  if (client->serverPort==-1) {
102  /* vncrec playing */
103  rfbVNCRec* rec = client->vncRec;
104  struct timeval tv;
105 
106  if (rec->readTimestamp) {
107  rec->readTimestamp = FALSE;
108  if (!fread(&tv,sizeof(struct timeval),1,rec->file))
109  return FALSE;
110 
111  tv.tv_sec = rfbClientSwap32IfLE (tv.tv_sec);
112  tv.tv_usec = rfbClientSwap32IfLE (tv.tv_usec);
113 
114  if (rec->tv.tv_sec!=0 && !rec->doNotSleep) {
115  struct timeval diff;
116  diff.tv_sec = tv.tv_sec - rec->tv.tv_sec;
117  diff.tv_usec = tv.tv_usec - rec->tv.tv_usec;
118  if(diff.tv_usec<0) {
119  diff.tv_sec--;
120  diff.tv_usec+=1000000;
121  }
122 #ifndef WIN32
123  sleep (diff.tv_sec);
124  usleep (diff.tv_usec);
125 #else
126  Sleep (diff.tv_sec * 1000 + diff.tv_usec/1000);
127 #endif
128  }
129 
130  rec->tv=tv;
131  }
132 
133  return (fread(out,1,n,rec->file) != n ? FALSE : TRUE);
134  }
135 
136  if (n <= client->buffered) {
137  memcpy(out, client->bufoutptr, n);
138  client->bufoutptr += n;
139  client->buffered -= n;
140 #ifdef DEBUG_READ_EXACT
141  goto hexdump;
142 #endif
143  return TRUE;
144  }
145 
146  memcpy(out, client->bufoutptr, client->buffered);
147 
148  out += client->buffered;
149  n -= client->buffered;
150 
151  client->bufoutptr = client->buf;
152  client->buffered = 0;
153 
154  if (n <= RFB_BUF_SIZE) {
155 
156  while (client->buffered < n) {
157  int i;
158  if (client->tlsSession)
159  i = ReadFromTLS(client, client->buf + client->buffered, RFB_BUF_SIZE - client->buffered);
160  else
161 #ifdef LIBVNCSERVER_HAVE_SASL
162  if (client->saslconn)
163  i = ReadFromSASL(client, client->buf + client->buffered, RFB_BUF_SIZE - client->buffered);
164  else {
165 #endif /* LIBVNCSERVER_HAVE_SASL */
166  i = read(client->sock, client->buf + client->buffered, RFB_BUF_SIZE - client->buffered);
167 #ifdef WIN32
168  if (i < 0) errno=WSAGetLastError();
169 #endif
170 #ifdef LIBVNCSERVER_HAVE_SASL
171  }
172 #endif
173 
174  if (i <= 0) {
175  if (i < 0) {
176  if (errno == EWOULDBLOCK || errno == EAGAIN) {
177  /* TODO:
178  ProcessXtEvents();
179  */
180  WaitForMessage(client, 100000);
181  i = 0;
182  } else {
183  rfbClientErr("read (%d: %s)\n",errno,strerror(errno));
184  return FALSE;
185  }
186  } else {
187  if (errorMessageOnReadFailure) {
188  rfbClientLog("VNC server closed connection\n");
189  }
190  return FALSE;
191  }
192  }
193  client->buffered += i;
194  }
195 
196  memcpy(out, client->bufoutptr, n);
197  client->bufoutptr += n;
198  client->buffered -= n;
199 
200  } else {
201 
202  while (n > 0) {
203  int i;
204  if (client->tlsSession)
205  i = ReadFromTLS(client, out, n);
206  else
207 #ifdef LIBVNCSERVER_HAVE_SASL
208  if (client->saslconn)
209  i = ReadFromSASL(client, out, n);
210  else
211 #endif
212  i = read(client->sock, out, n);
213 
214  if (i <= 0) {
215  if (i < 0) {
216 #ifdef WIN32
217  errno=WSAGetLastError();
218 #endif
219  if (errno == EWOULDBLOCK || errno == EAGAIN) {
220  /* TODO:
221  ProcessXtEvents();
222  */
223  WaitForMessage(client, 100000);
224  i = 0;
225  } else {
226  rfbClientErr("read (%s)\n",strerror(errno));
227  return FALSE;
228  }
229  } else {
230  if (errorMessageOnReadFailure) {
231  rfbClientLog("VNC server closed connection\n");
232  }
233  return FALSE;
234  }
235  }
236  out += i;
237  n -= i;
238  }
239  }
240 
241 #ifdef DEBUG_READ_EXACT
242 hexdump:
243  { int ii;
244  for(ii=0;ii<nn;ii++)
245  fprintf(stderr,"%02x ",(unsigned char)oout[ii]);
246  fprintf(stderr,"\n");
247  }
248 #endif
249 
250  return TRUE;
251 }

258 rfbBool
259 WriteToRFBServer(rfbClient* client, char *buf, int n)
260 {
261  fd_set fds;
262  int i = 0;
263  int j;
264  const char *obuf = buf;
265 #ifdef LIBVNCSERVER_HAVE_SASL
266  const char *output;
267  unsigned int outputlen;
268  int err;
269 #endif /* LIBVNCSERVER_HAVE_SASL */
270 
271  if (client->serverPort==-1)
272  return TRUE; /* vncrec playing */
273 
274  if (client->tlsSession) {
275  /* WriteToTLS() will guarantee either everything is written, or error/eof returns */
276  i = WriteToTLS(client, buf, n);
277  if (i <= 0) return FALSE;
278 
279  return TRUE;
280  }
281 #ifdef LIBVNCSERVER_HAVE_SASL
282  if (client->saslconn) {
283  err = sasl_encode(client->saslconn,
284  buf, n,
285  &output, &outputlen);
286  if (err != SASL_OK) {
287  rfbClientLog("Failed to encode SASL data %s",
288  sasl_errstring(err, NULL, NULL));
289  return FALSE;
290  }
291  obuf = output;
292  n = outputlen;
293  }
294 #endif /* LIBVNCSERVER_HAVE_SASL */
295 
296  while (i < n) {
297  j = write(client->sock, obuf + i, (n - i));
298  if (j <= 0) {
299  if (j < 0) {
300 #ifdef WIN32
301  errno=WSAGetLastError();
302 #endif
303  if (errno == EWOULDBLOCK ||
304 #ifdef LIBVNCSERVER_ENOENT_WORKAROUND
305  errno == ENOENT ||
306 #endif
307  errno == EAGAIN) {
308  FD_ZERO(&fds);
309  FD_SET(client->sock,&fds);
310 
311  if (select(client->sock+1, NULL, &fds, NULL, NULL) <= 0) {
312  rfbClientErr("select\n");
313  return FALSE;
314  }
315  j = 0;
316  } else {
317  rfbClientErr("write\n");
318  return FALSE;
319  }
320  } else {
321  rfbClientLog("write failed\n");
322  return FALSE;
323  }
324  }
325  i += j;
326  }
327  return TRUE;
328 }

530  * ListenAtTcpPortAndAddress starts listening at the given TCP port on
531  * the given IP address
532  */
533 
534 int
535 ListenAtTcpPortAndAddress(int port, const char *address)
536 {
537  int sock;
538  int one = 1;
539 #ifndef LIBVNCSERVER_IPv6
540  struct sockaddr_in addr;
541 
542  addr.sin_family = AF_INET;
543  addr.sin_port = htons(port);
544  if (address) {
545  addr.sin_addr.s_addr = inet_addr(address);
546  } else {
547  addr.sin_addr.s_addr = htonl(INADDR_ANY);
548  }
549 
550  if (!initSockets())
551  return -1;
552 
553  sock = socket(AF_INET, SOCK_STREAM, 0);
554  if (sock < 0) {
555  rfbClientErr("ListenAtTcpPort: socket\n");
556  return -1;
557  }
558 
559  if (setsockopt(sock, SOL_SOCKET, SO_REUSEADDR,
560  (const char *)&one, sizeof(one)) < 0) {
561  rfbClientErr("ListenAtTcpPort: setsockopt\n");
562  close(sock);
563  return -1;
564  }
565 
566  if (bind(sock, (struct sockaddr *)&addr, sizeof(addr)) < 0) {
567  rfbClientErr("ListenAtTcpPort: bind\n");
568  close(sock);
569  return -1;
570  }
571 
572 #else
573  int rv;
574  struct addrinfo hints, *servinfo, *p;
575  char port_str[8];
576 
577  snprintf(port_str, 8, "%d", port);
578 
579  memset(&hints, 0, sizeof(hints));
580  hints.ai_family = AF_UNSPEC;
581  hints.ai_socktype = SOCK_STREAM;
582  hints.ai_flags = AI_PASSIVE; /* fill in wildcard address if address == NULL */
583 
584  if (!initSockets())
585  return -1;
586 
587  if ((rv = getaddrinfo(address, port_str, &hints, &servinfo)) != 0) {
588  rfbClientErr("ListenAtTcpPortAndAddress: error in getaddrinfo: %s\n", gai_strerror(rv));
589  return -1;
590  }
591 
592  /* loop through all the results and bind to the first we can */
593  for(p = servinfo; p != NULL; p = p->ai_next) {
594  if ((sock = socket(p->ai_family, p->ai_socktype, p->ai_protocol)) < 0) {
595  continue;
596  }
597 
598 #ifdef IPV6_V6ONLY
599  /* we have separate IPv4 and IPv6 sockets since some OS's do not support dual binding */
600  if (p->ai_family == AF_INET6 && setsockopt(sock, IPPROTO_IPV6, IPV6_V6ONLY, (char *)&one, sizeof(one)) < 0) {
601  rfbClientErr("ListenAtTcpPortAndAddress: error in setsockopt IPV6_V6ONLY: %s\n", strerror(errno));
602  close(sock);
603  freeaddrinfo(servinfo);
604  return -1;
605  }
606 #endif
607 
608  if (setsockopt(sock, SOL_SOCKET, SO_REUSEADDR, (char *)&one, sizeof(one)) < 0) {
609  rfbClientErr("ListenAtTcpPortAndAddress: error in setsockopt SO_REUSEADDR: %s\n", strerror(errno));
610  close(sock);
611  freeaddrinfo(servinfo);
612  return -1;
613  }
614 
615  if (bind(sock, p->ai_addr, p->ai_addrlen) < 0) {
616  close(sock);
617  continue;
618  }
619 
620  break;
621  }
622 
623  if (p == NULL) {
624  rfbClientErr("ListenAtTcpPortAndAddress: error in bind: %s\n", strerror(errno));
625  return -1;
626  }
627 
628  /* all done with this structure now */
629  freeaddrinfo(servinfo);
630 #endif
631 
632  if (listen(sock, 5) < 0) {
633  rfbClientErr("ListenAtTcpPort: listen\n");
634  close(sock);
635  return -1;
636  }
637 
638  return sock;
639 }
