57 #define RGB_TO_PIXEL(bpp,r,g,b) \
58  (((CARD##bpp)(r) & client->format.redMax) << client->format.redShift | \
59  ((CARD##bpp)(g) & client->format.greenMax) << client->format.greenShift | \
60  ((CARD##bpp)(b) & client->format.blueMax) << client->format.blueShift)
61 
62 #define RGB24_TO_PIXEL(bpp,r,g,b) \
63  ((((CARD##bpp)(r) & 0xFF) * client->format.redMax + 127) / 255 \
64  << client->format.redShift | \
65  (((CARD##bpp)(g) & 0xFF) * client->format.greenMax + 127) / 255 \
66  << client->format.greenShift | \
67  (((CARD##bpp)(b) & 0xFF) * client->format.blueMax + 127) / 255 \
68  << client->format.blueShift)
69 
70 #define RGB24_TO_PIXEL32(r,g,b) \
71  (((uint32_t)(r) & 0xFF) << client->format.redShift | \
72  ((uint32_t)(g) & 0xFF) << client->format.greenShift | \
73  ((uint32_t)(b) & 0xFF) << client->format.blueShift)
74 
75 #endif

96 static rfbBool
97 HandleTightBPP (rfbClient* client, int rx, int ry, int rw, int rh)
98 {
99  CARDBPP fill_colour;
100  uint8_t comp_ctl;
101  uint8_t filter_id;
102  filterPtrBPP filterFn;
103  z_streamp zs;
104  int err, stream_id, compressedLen, bitsPixel;
105  int bufferSize, rowSize, numRows, portionLen, rowsProcessed, extraBytes;
106  rfbBool readUncompressed = FALSE;
107 
108  if (client->frameBuffer == NULL)
109  return FALSE;
110 
111  if (rx + rw > client->width || ry + rh > client->height) {
112  rfbClientLog("Rect out of bounds: %dx%d at (%d, %d)\n", rx, ry, rw, rh);
113  return FALSE;
114  }
115 
116  if (!ReadFromRFBServer(client, (char *)&comp_ctl, 1))
117  return FALSE;
118 
119  /* Flush zlib streams if we are told by the server to do so. */
120  for (stream_id = 0; stream_id < 4; stream_id++) {
121  if ((comp_ctl & 1) && client->zlibStreamActive[stream_id]) {
122  if (inflateEnd (&client->zlibStream[stream_id]) != Z_OK &&
123  client->zlibStream[stream_id].msg != NULL)
124  rfbClientLog("inflateEnd: %s\n", client->zlibStream[stream_id].msg);
125  client->zlibStreamActive[stream_id] = FALSE;
126  }
127  comp_ctl >>= 1;
128  }
129 
130  if ((comp_ctl & rfbTightNoZlib) == rfbTightNoZlib) {
131  comp_ctl &= ~(rfbTightNoZlib);
132  readUncompressed = TRUE;
133  }
134 
135  /* Handle solid rectangles. */
136  if (comp_ctl == rfbTightFill) {
137 #if BPP == 32
138  if (client->format.depth == 24 && client->format.redMax == 0xFF &&
139  client->format.greenMax == 0xFF && client->format.blueMax == 0xFF) {
140  if (!ReadFromRFBServer(client, client->buffer, 3))
141  return FALSE;
142  fill_colour = RGB24_TO_PIXEL32(client->buffer[0], client->buffer[1], client->buffer[2]);
143  } else {
144  if (!ReadFromRFBServer(client, (char*)&fill_colour, sizeof(fill_colour)))
145  return FALSE;
146  }
147 #else
148  if (!ReadFromRFBServer(client, (char*)&fill_colour, sizeof(fill_colour)))
149  return FALSE;
150 #endif
151 
152  client->GotFillRect(client, rx, ry, rw, rh, fill_colour);
153 
154  return TRUE;
155  }
156 
157 #if BPP == 8
158  if (comp_ctl == rfbTightJpeg) {
159  rfbClientLog("Tight encoding: JPEG is not supported in 8 bpp mode.\n");
160  return FALSE;
161  }
162 #else
163  if (comp_ctl == rfbTightJpeg) {
164  return DecompressJpegRectBPP(client, rx, ry, rw, rh);
165  }
166 #endif
167 
168  /* Quit on unsupported subencoding value. */
169  if (comp_ctl > rfbTightMaxSubencoding) {
170  rfbClientLog("Tight encoding: bad subencoding value received.\n");
171  return FALSE;
172  }
173 
174  /*
175  * Here primary compression mode handling begins.
176  * Data was processed with optional filter + zlib compression.
177  */
178 
179  /* First, we should identify a filter to use. */
180  if ((comp_ctl & rfbTightExplicitFilter) != 0) {
181  if (!ReadFromRFBServer(client, (char*)&filter_id, 1))
182  return FALSE;
183 
184  switch (filter_id) {
185  case rfbTightFilterCopy:
186  filterFn = FilterCopyBPP;
187  bitsPixel = InitFilterCopyBPP(client, rw, rh);
188  break;
189  case rfbTightFilterPalette:
190  filterFn = FilterPaletteBPP;
191  bitsPixel = InitFilterPaletteBPP(client, rw, rh);
192  break;
193  case rfbTightFilterGradient:
194  filterFn = FilterGradientBPP;
195  bitsPixel = InitFilterGradientBPP(client, rw, rh);
196  break;
197  default:
198  rfbClientLog("Tight encoding: unknown filter code received.\n");
199  return FALSE;
200  }
201  } else {
202  filterFn = FilterCopyBPP;
203  bitsPixel = InitFilterCopyBPP(client, rw, rh);
204  }
205  if (bitsPixel == 0) {
206  rfbClientLog("Tight encoding: error receiving palette.\n");
207  return FALSE;
208  }
209 
210  /* Determine if the data should be decompressed or just copied. */
211  rowSize = (rw * bitsPixel + 7) / 8;
212  if (rh * rowSize < TIGHT_MIN_TO_COMPRESS) {
213  if (!ReadFromRFBServer(client, (char*)client->buffer, rh * rowSize))
214  return FALSE;
215 
216  filterFn(client, rx, ry, rh);
217 
218  return TRUE;
219  }
220 
221  /* Read the length (1..3 bytes) of compressed data following. */
222  compressedLen = (int)ReadCompactLen(client);
223  if (compressedLen <= 0) {
224  rfbClientLog("Incorrect data received from the server.\n");
225  return FALSE;
226  }
227  if (readUncompressed) {
228  if (!ReadFromRFBServer(client, (char*)client->buffer, compressedLen))
229  return FALSE;
230 
231  filterFn(client, rx, ry, rh);
232 
233  return TRUE;
234  }
235 
236  /* Now let's initialize compression stream if needed. */
237  stream_id = comp_ctl & 0x03;
238  zs = &client->zlibStream[stream_id];
239  if (!client->zlibStreamActive[stream_id]) {
240  zs->zalloc = Z_NULL;
241  zs->zfree = Z_NULL;
242  zs->opaque = Z_NULL;
243  err = inflateInit(zs);
244  if (err != Z_OK) {
245  if (zs->msg != NULL)
246  rfbClientLog("InflateInit error: %s.\n", zs->msg);
247  return FALSE;
248  }
249  client->zlibStreamActive[stream_id] = TRUE;
250  }
251 
252  /* Read, decode and draw actual pixel data in a loop. */
253 
254  bufferSize = RFB_BUFFER_SIZE * bitsPixel / (bitsPixel + BPP) & 0xFFFFFFFC;
255  if (rowSize > bufferSize) {
256  /* Should be impossible when RFB_BUFFER_SIZE >= 16384 */
257  rfbClientLog("Internal error: incorrect buffer size.\n");
258  return FALSE;
259  }
260 
261  rowsProcessed = 0;
262  extraBytes = 0;
263 
264  while (compressedLen > 0) {
265  if (compressedLen > ZLIB_BUFFER_SIZE)
266  portionLen = ZLIB_BUFFER_SIZE;
267  else
268  portionLen = compressedLen;
269 
270  if (!ReadFromRFBServer(client, (char*)client->zlib_buffer, portionLen))
271  return FALSE;
272 
273  compressedLen -= portionLen;
274 
275  zs->next_in = (Bytef *)client->zlib_buffer;
276  zs->avail_in = portionLen;
277 
278  do {
279  zs->next_out = (Bytef *)&client->buffer[extraBytes];
280  zs->avail_out = bufferSize - extraBytes;
281 
282  err = inflate(zs, Z_SYNC_FLUSH);
283  if (err == Z_BUF_ERROR) /* Input exhausted -- no problem. */
284  break;
285  if (err != Z_OK && err != Z_STREAM_END) {
286  if (zs->msg != NULL) {
287  rfbClientLog("Inflate error: %s.\n", zs->msg);
288  } else {
289  rfbClientLog("Inflate error: %d.\n", err);
290  }
291  return FALSE;
292  }
293 
294  numRows = (bufferSize - zs->avail_out) / rowSize;
295 
296  filterFn(client, rx, ry+rowsProcessed, numRows);
297 
298  extraBytes = bufferSize - zs->avail_out - numRows * rowSize;
299  if (extraBytes > 0)
300  memcpy(client->buffer, &client->buffer[numRows * rowSize], extraBytes);
301 
302  rowsProcessed += numRows;
303  }
304  while (zs->avail_out == 0);
305  }
306 
307  if (rowsProcessed != rh) {
308  rfbClientLog("Incorrect number of scan lines after decompression.\n");
309  return FALSE;
310  }
311 
312  return TRUE;
313 }

340 FilterCopyBPP (rfbClient* client, int srcx, int srcy, int numRows)
341 {
342  CARDBPP *dst =
343  (CARDBPP *)&client->frameBuffer[(srcy * client->width + srcx) * BPP / 8];
344  int y;
345 
346 #if BPP == 32
347  int x;
348 
349  if (client->cutZeros) {
350  for (y = 0; y < numRows; y++) {
351  for (x = 0; x < client->rectWidth; x++) {
352  dst[y*client->width+x] =
353  RGB24_TO_PIXEL32(client->buffer[(y*client->rectWidth+x)*3],
354  client->buffer[(y*client->rectWidth+x)*3+1],
355  client->buffer[(y*client->rectWidth+x)*3+2]);
356  }
357  }
358  return;
359  }
360 #endif
361 
362  for (y = 0; y < numRows; y++)
363  memcpy (&dst[y*client->width],
364  &client->buffer[y * client->rectWidth * (BPP / 8)],
365  client->rectWidth * (BPP / 8));

384 static void
385 FilterGradient24 (rfbClient* client, int srcx, int srcy, int numRows)
386 {
387  CARDBPP *dst =
388  (CARDBPP *)&client->frameBuffer[(srcy * client->width + srcx) * BPP / 8];
389  int x, y, c;
390  uint8_t thisRow[2048*3];
391  uint8_t pix[3];
392  int est[3];
393 
394  for (y = 0; y < numRows; y++) {
395 
396  /* First pixel in a row */
397  for (c = 0; c < 3; c++) {
398  pix[c] = client->tightPrevRow[c] + client->buffer[y*client->rectWidth*3+c];
399  thisRow[c] = pix[c];
400  }
401  dst[y*client->width] = RGB24_TO_PIXEL32(pix[0], pix[1], pix[2]);
402 
403  /* Remaining pixels of a row */
404  for (x = 1; x < client->rectWidth; x++) {
405  for (c = 0; c < 3; c++) {
406  est[c] = (int)client->tightPrevRow[x*3+c] + (int)pix[c] -
407  (int)client->tightPrevRow[(x-1)*3+c];
408  if (est[c] > 0xFF) {
409  est[c] = 0xFF;
410  } else if (est[c] < 0x00) {
411  est[c] = 0x00;
412  }
413  pix[c] = (uint8_t)est[c] + client->buffer[(y*client->rectWidth+x)*3+c];
414  thisRow[x*3+c] = pix[c];
415  }
416  dst[y*client->width+x] = RGB24_TO_PIXEL32(pix[0], pix[1], pix[2]);
417  }
418 
419  memcpy(client->tightPrevRow, thisRow, client->rectWidth * 3);
420  }
421 }

425 static void
426 FilterGradientBPP (rfbClient* client, int srcx, int srcy, int numRows)
427 {
428  CARDBPP *dst =
429  (CARDBPP *)&client->frameBuffer[(srcy * client->width + srcx) * BPP / 8];
430  int x, y, c;
431  CARDBPP *src = (CARDBPP *)client->buffer;
432  uint16_t *thatRow = (uint16_t *)client->tightPrevRow;
433  uint16_t thisRow[2048*3];
434  uint16_t pix[3];
435  uint16_t max[3];
436  int shift[3];
437  int est[3];
438 
439 #if BPP == 32
440  if (client->cutZeros) {
441  FilterGradient24(client, srcx, srcy, numRows);
442  return;
443  }
444 #endif
445 
446  max[0] = client->format.redMax;
447  max[1] = client->format.greenMax;
448  max[2] = client->format.blueMax;
449 
450  shift[0] = client->format.redShift;
451  shift[1] = client->format.greenShift;
452  shift[2] = client->format.blueShift;
453 
454  for (y = 0; y < numRows; y++) {
455 
456  /* First pixel in a row */
457  for (c = 0; c < 3; c++) {
458  pix[c] = (uint16_t)(((src[y*client->rectWidth] >> shift[c]) + thatRow[c]) & max[c]);
459  thisRow[c] = pix[c];
460  }
461  dst[y*client->width] = RGB_TO_PIXEL(BPP, pix[0], pix[1], pix[2]);
462 
463  /* Remaining pixels of a row */
464  for (x = 1; x < client->rectWidth; x++) {
465  for (c = 0; c < 3; c++) {
466  est[c] = (int)thatRow[x*3+c] + (int)pix[c] - (int)thatRow[(x-1)*3+c];
467  if (est[c] > (int)max[c]) {
468  est[c] = (int)max[c];
469  } else if (est[c] < 0) {
470  est[c] = 0;
471  }
472  pix[c] = (uint16_t)(((src[y*client->rectWidth+x] >> shift[c]) + est[c]) & max[c]);
473  thisRow[x*3+c] = pix[c];
474  }
475  dst[y*client->width+x] = RGB_TO_PIXEL(BPP, pix[0], pix[1], pix[2]);
476  }
477  memcpy(thatRow, thisRow, client->rectWidth * 3 * sizeof(uint16_t));
478  }
479 }

520 FilterPaletteBPP (rfbClient* client, int srcx, int srcy, int numRows)
521 {
522  int x, y, b, w;
523  CARDBPP *dst =
524  (CARDBPP *)&client->frameBuffer[(srcy * client->width + srcx) * BPP / 8];
525  uint8_t *src = (uint8_t *)client->buffer;
526  CARDBPP *palette = (CARDBPP *)client->tightPalette;
527 
528  if (client->rectColors == 2) {
529  w = (client->rectWidth + 7) / 8;
530  for (y = 0; y < numRows; y++) {
531  for (x = 0; x < client->rectWidth / 8; x++) {
532  for (b = 7; b >= 0; b--)
533  dst[y*client->width+x*8+7-b] = palette[src[y*w+x] >> b & 1];
534  }
535  for (b = 7; b >= 8 - client->rectWidth % 8; b--) {
536  dst[y*client->width+x*8+7-b] = palette[src[y*w+x] >> b & 1];
537  }
538  }
539  } else {
540  for (y = 0; y < numRows; y++)
541  for (x = 0; x < client->rectWidth; x++)
542  dst[y*client->width+x] = palette[(int)src[y*client->rectWidth+x]];
543  }

554 static rfbBool
555 DecompressJpegRectBPP(rfbClient* client, int x, int y, int w, int h)
556 {
557  int compressedLen;
558  uint8_t *compressedData, *dst;
559  int pixelSize, pitch, flags = 0;
560 
561  compressedLen = (int)ReadCompactLen(client);
562  if (compressedLen <= 0) {
563  rfbClientLog("Incorrect data received from the server.\n");
564  return FALSE;
565  }
566 
567  compressedData = malloc(compressedLen);
568  if (compressedData == NULL) {
569  rfbClientLog("Memory allocation error.\n");
570  return FALSE;
571  }
572 
573  if (!ReadFromRFBServer(client, (char*)compressedData, compressedLen)) {
574  free(compressedData);
575  return FALSE;
576  }
577 
578  if(client->GotJpeg != NULL)
579  return client->GotJpeg(client, compressedData, compressedLen, x, y, w, h);
580 
581  if (!client->tjhnd) {
582  if ((client->tjhnd = tjInitDecompress()) == NULL) {
583  rfbClientLog("TurboJPEG error: %s\n", tjGetErrorStr());
584  free(compressedData);
585  return FALSE;
586  }
587  }
588 
589 #if BPP == 16
590  flags = 0;
591  pixelSize = 3;
592  pitch = w * pixelSize;
593  dst = (uint8_t *)client->buffer;
594 #else
595  if (client->format.bigEndian) flags |= TJ_ALPHAFIRST;
596  if (client->format.redShift == 16 && client->format.blueShift == 0)
597  flags |= TJ_BGR;
598  if (client->format.bigEndian) flags ^= TJ_BGR;
599  pixelSize = BPP / 8;
600  pitch = client->width * pixelSize;
601  dst = &client->frameBuffer[y * pitch + x * pixelSize];
602 #endif
603 
604  if (tjDecompress(client->tjhnd, compressedData, (unsigned long)compressedLen,
605  dst, w, pitch, h, pixelSize, flags)==-1) {
606  rfbClientLog("TurboJPEG error: %s\n", tjGetErrorStr());
607  free(compressedData);
608  return FALSE;
609  }
610 
611  free(compressedData);
612 
613 #if BPP == 16
614  pixelSize = BPP / 8;
615  pitch = client->width * pixelSize;
616  dst = &client->frameBuffer[y * pitch + x * pixelSize];
617  {
618  CARDBPP *dst16=(CARDBPP *)dst, *dst2;
619  char *src = client->buffer;
620  int i, j;
621 
622  for (j = 0; j < h; j++) {
623  for (i = 0, dst2 = dst16; i < w; i++, dst2++, src += 3) {
624  *dst2 = RGB24_TO_PIXEL(BPP, src[0], src[1], src[2]);
625  }
626  dst16 += client->width;
627  }
628  }
629 #endif
630 
631  return TRUE;
632 }

636 static long
637 ReadCompactLen (rfbClient* client)
638 {
639  long len;
