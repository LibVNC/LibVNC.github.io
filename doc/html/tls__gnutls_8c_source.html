40 static int
41 verify_certificate_callback (gnutls_session_t session)
42 {
43  unsigned int status;
44  const gnutls_datum_t *cert_list;
45  unsigned int cert_list_size;
46  int ret;
47  gnutls_x509_crt_t cert;
48  rfbClient *sptr;
49  char *hostname;
50 
51  sptr = (rfbClient *)gnutls_session_get_ptr(session);
52  if (!sptr) {
53  rfbClientLog("Failed to validate certificate - missing client data\n");
54  return GNUTLS_E_CERTIFICATE_ERROR;
55  }
56 
57  hostname = sptr->serverHost;
58  if (!hostname) {
59  rfbClientLog("No server hostname found for client\n");
60  return GNUTLS_E_CERTIFICATE_ERROR;
61  }
62 
63  /* This verification function uses the trusted CAs in the credentials
64  * structure. So you must have installed one or more CA certificates.
65  */
66  ret = gnutls_certificate_verify_peers2 (session, &status);
67  if (ret < 0)
68  {
69  rfbClientLog ("Certificate validation call failed\n");
70  return GNUTLS_E_CERTIFICATE_ERROR;
71  }
72 
73  if (status & GNUTLS_CERT_INVALID)
74  rfbClientLog("The certificate is not trusted.\n");
75 
76  if (status & GNUTLS_CERT_SIGNER_NOT_FOUND)
77  rfbClientLog("The certificate hasn't got a known issuer.\n");
78 
79  if (status & GNUTLS_CERT_REVOKED)
80  rfbClientLog("The certificate has been revoked.\n");
81 
82  if (status & GNUTLS_CERT_EXPIRED)
83  rfbClientLog("The certificate has expired\n");
84 
85  if (status & GNUTLS_CERT_NOT_ACTIVATED)
86  rfbClientLog("The certificate is not yet activated\n");
87 
88  if (status)
89  return GNUTLS_E_CERTIFICATE_ERROR;
90 
91  /* Up to here the process is the same for X.509 certificates and
92  * OpenPGP keys. From now on X.509 certificates are assumed. This can
93  * be easily extended to work with openpgp keys as well.
94  */
95  if (gnutls_certificate_type_get (session) != GNUTLS_CRT_X509) {
96  rfbClientLog("The certificate was not X509\n");
97  return GNUTLS_E_CERTIFICATE_ERROR;
98  }
99 
100  if (gnutls_x509_crt_init (&cert) < 0)
101  {
102  rfbClientLog("Error initialising certificate structure\n");
103  return GNUTLS_E_CERTIFICATE_ERROR;
104  }
105 
106  cert_list = gnutls_certificate_get_peers (session, &cert_list_size);
107  if (cert_list == NULL)
108  {
109  rfbClientLog("No certificate was found!\n");
110  return GNUTLS_E_CERTIFICATE_ERROR;
111  }
112 
113  if (gnutls_x509_crt_import (cert, &cert_list[0], GNUTLS_X509_FMT_DER) < 0)
114  {
115  rfbClientLog("Error parsing certificate\n");
116  return GNUTLS_E_CERTIFICATE_ERROR;
117  }
118 
119  if (!gnutls_x509_crt_check_hostname (cert, hostname))
120  {
121  rfbClientLog("The certificate's owner does not match hostname '%s'\n",
122  hostname);
123  return GNUTLS_E_CERTIFICATE_ERROR;
124  }
125 
126  gnutls_x509_crt_deinit (cert);
127 
128  /* notify gnutls to continue handshake normally */
129  return 0;
130 }

150 /*
151  * On Windows, translate WSAGetLastError() to errno values as GNU TLS does it
152  * internally too. This is necessary because send() and recv() on Windows
153  * don't set errno when they fail but GNUTLS expects a proper errno value.
154  *
155  * Use gnutls_transport_set_global_errno() like the GNU TLS documentation
156  * suggests to avoid problems with different errno variables when GNU TLS and
157  * libvncclient are linked to different versions of msvcrt.dll.
158  */
159 #ifdef WIN32
160 static void WSAtoTLSErrno(gnutls_session_t* session)
161 {
162  switch(WSAGetLastError()) {
163 #if (GNUTLS_VERSION_NUMBER >= 0x029901)
164  case WSAEWOULDBLOCK:
165  gnutls_transport_set_errno(session, EAGAIN);
166  break;
167  case WSAEINTR:
168  gnutls_transport_set_errno(session, EINTR);
169  break;
170  default:
171  gnutls_transport_set_errno(session, EIO);
172  break;
173 #else
174  case WSAEWOULDBLOCK:
175  gnutls_transport_set_global_errno(EAGAIN);
176  break;
177  case WSAEINTR:
178  gnutls_transport_set_global_errno(EINTR);
179  break;
180  default:
181  gnutls_transport_set_global_errno(EIO);
182  break;
183 #endif
184  }
185 }
186 #endif

188 static ssize_t
189 PushTLS(gnutls_transport_ptr_t transport, const void *data, size_t len)
190 {
191  rfbClient *client = (rfbClient*)transport;
192  int ret;
193 
194  while (1)
195  {
196  ret = write(client->sock, data, len);
197  if (ret < 0)
198  {
199 #ifdef WIN32
200  WSAtoTLSErrno((gnutls_session_t*)&client->tlsSession);
201 #endif
202  if (errno == EINTR) continue;
203  return -1;
204  }
205  return ret;
206  }

231 static rfbBool
232 InitializeTLSSession(rfbClient* client, rfbBool anonTLS)
233 {
234  int ret;
235  const char *p;
236 
237  if (client->tlsSession) return TRUE;
238 
239  if ((ret = gnutls_init((gnutls_session_t*)&client->tlsSession, GNUTLS_CLIENT)) < 0)
240  {
241  rfbClientLog("Failed to initialized TLS session: %s.\n", gnutls_strerror(ret));
242  return FALSE;
243  }
244 
245  if ((ret = gnutls_priority_set_direct((gnutls_session_t)client->tlsSession,
246  anonTLS ? rfbAnonTLSPriority : rfbTLSPriority, &p)) < 0)
247  {
248  rfbClientLog("Warning: Failed to set TLS priority: %s (%s).\n", gnutls_strerror(ret), p);
249  }
250 
251  gnutls_transport_set_ptr((gnutls_session_t)client->tlsSession, (gnutls_transport_ptr_t)client);
252  gnutls_transport_set_push_function((gnutls_session_t)client->tlsSession, PushTLS);
253  gnutls_transport_set_pull_function((gnutls_session_t)client->tlsSession, PullTLS);
254 
255  INIT_MUTEX(client->tlsRwMutex);
256 
257  rfbClientLog("TLS session initialized.\n");
258 
259  return TRUE;
260 }

279 static rfbBool
280 HandshakeTLS(rfbClient* client)
281 {
282  int timeout = 15;
283  int ret;
284 
285  while (timeout > 0 && (ret = gnutls_handshake((gnutls_session_t)client->tlsSession)) < 0)
286  {
287  if (!gnutls_error_is_fatal(ret))
288  {
289  rfbClientLog("TLS handshake blocking.\n");
290  sleep(1);
291  timeout--;
292  continue;
293  }
294  rfbClientLog("TLS handshake failed: %s.\n", gnutls_strerror(ret));
295 
296  FreeTLS(client);
297  return FALSE;
298  }
299 
300  if (timeout <= 0)
301  {
302  rfbClientLog("TLS handshake timeout.\n");
303  FreeTLS(client);
304  return FALSE;
305  }
306 
307  rfbClientLog("TLS handshake done.\n");
308  return TRUE;
309 }

311 /* VeNCrypt sub auth. 1 byte auth count, followed by count * 4 byte integers */
312 static rfbBool
313 ReadVeNCryptSecurityType(rfbClient* client, uint32_t *result)
314 {
315  uint8_t count=0;
316  uint8_t loop=0;
317  uint8_t flag=0;
318  uint32_t tAuth[256], t;
319  char buf1[500],buf2[10];
320  uint32_t authScheme;
321 
322  if (!ReadFromRFBServer(client, (char *)&count, 1)) return FALSE;
323 
324  if (count==0)
325  {
326  rfbClientLog("List of security types is ZERO. Giving up.\n");
327  return FALSE;
328  }
329 
330  rfbClientLog("We have %d security types to read\n", count);
331  authScheme=0;
332  /* now, we have a list of available security types to read ( uint8_t[] ) */
333  for (loop=0;loop<count;loop++)
334  {
335  if (!ReadFromRFBServer(client, (char *)&tAuth[loop], 4)) return FALSE;
336  t=rfbClientSwap32IfLE(tAuth[loop]);
337  rfbClientLog("%d) Received security type %d\n", loop, t);
338  if (flag) continue;
339  if (t==rfbVeNCryptTLSNone ||
340  t==rfbVeNCryptTLSVNC ||
341  t==rfbVeNCryptTLSPlain ||
342 #ifdef LIBVNCSERVER_HAVE_SASL
343  t==rfbVeNCryptTLSSASL ||
344  t==rfbVeNCryptX509SASL ||
345 #endif /*LIBVNCSERVER_HAVE_SASL */
346  t==rfbVeNCryptX509None ||
347  t==rfbVeNCryptX509VNC ||
348  t==rfbVeNCryptX509Plain)
349  {
350  flag++;
351  authScheme=t;
352  rfbClientLog("Selecting security type %d (%d/%d in the list)\n", authScheme, loop, count);
353  /* send back 4 bytes (in original byte order!) indicating which security type to use */
354  if (!WriteToRFBServer(client, (char *)&tAuth[loop], 4)) return FALSE;
355  }
356  tAuth[loop]=t;
357  }
358  if (authScheme==0)
359  {
360  memset(buf1, 0, sizeof(buf1));
361  for (loop=0;loop<count;loop++)
362  {
363  if (strlen(buf1)>=sizeof(buf1)-1) break;
364  snprintf(buf2, sizeof(buf2), (loop>0 ? ", %d" : "%d"), (int)tAuth[loop]);
365  strncat(buf1, buf2, sizeof(buf1)-strlen(buf1)-1);
366  }
367  rfbClientLog("Unknown VeNCrypt authentication scheme from VNC server: %s\n",
368  buf1);
369  return FALSE;
370  }
371  *result = authScheme;
372  return TRUE;
373 }

385 static gnutls_certificate_credentials_t
386 CreateX509CertCredential(rfbCredential *cred)
387 {
388  gnutls_certificate_credentials_t x509_cred;
389  int ret;
390 
391  if (!cred->x509Credential.x509CACertFile)
392  {
393  rfbClientLog("No CA certificate provided.\n");
394  return NULL;
395  }
396 
397  if ((ret = gnutls_certificate_allocate_credentials(&x509_cred)) < 0)
398  {
399  rfbClientLog("Cannot allocate credentials: %s.\n", gnutls_strerror(ret));
400  return NULL;
401  }
402  if ((ret = gnutls_certificate_set_x509_trust_file(x509_cred,
403  cred->x509Credential.x509CACertFile, GNUTLS_X509_FMT_PEM)) < 0)
404  {
405  rfbClientLog("Cannot load CA credentials: %s.\n", gnutls_strerror(ret));
406  gnutls_certificate_free_credentials (x509_cred);
407  return NULL;
408  }
409  if (cred->x509Credential.x509ClientCertFile && cred->x509Credential.x509ClientKeyFile)
410  {
411  if ((ret = gnutls_certificate_set_x509_key_file(x509_cred,
412  cred->x509Credential.x509ClientCertFile, cred->x509Credential.x509ClientKeyFile,
413  GNUTLS_X509_FMT_PEM)) < 0)
414  {
415  rfbClientLog("Cannot load client certificate or key: %s.\n", gnutls_strerror(ret));
416  gnutls_certificate_free_credentials (x509_cred);
417  return NULL;
418  }
419  } else
420  {
421  rfbClientLog("No client certificate or key provided.\n");
422  }
423  if (cred->x509Credential.x509CACrlFile)
424  {
425  if ((ret = gnutls_certificate_set_x509_crl_file(x509_cred,
426  cred->x509Credential.x509CACrlFile, GNUTLS_X509_FMT_PEM)) < 0)
427  {
428  rfbClientLog("Cannot load CRL: %s.\n", gnutls_strerror(ret));
429  gnutls_certificate_free_credentials (x509_cred);
430  return NULL;
431  }
432  } else
433  {
434  rfbClientLog("No CRL provided.\n");
435  }
436  gnutls_certificate_set_dh_params (x509_cred, rfbDHParams);
437  return x509_cred;
438 }

453 rfbBool
454 HandleVeNCryptAuth(rfbClient* client)
455 {
456  uint8_t major, minor, status;
457  uint32_t authScheme;
458  rfbBool anonTLS;
459  gnutls_certificate_credentials_t x509_cred = NULL;
460  int ret;
461 
462  if (!InitializeTLS()) return FALSE;
463 
464  /* Read VeNCrypt version */
465  if (!ReadFromRFBServer(client, (char *)&major, 1) ||
466  !ReadFromRFBServer(client, (char *)&minor, 1))
467  {
468  return FALSE;
469  }
470  rfbClientLog("Got VeNCrypt version %d.%d from server.\n", (int)major, (int)minor);
471 
472  if (major != 0 && minor != 2)
473  {
474  rfbClientLog("Unsupported VeNCrypt version.\n");
475  return FALSE;
476  }
477 
478  if (!WriteToRFBServer(client, (char *)&major, 1) ||
479  !WriteToRFBServer(client, (char *)&minor, 1) ||
480  !ReadFromRFBServer(client, (char *)&status, 1))
481  {
482  return FALSE;
483  }
484 
485  if (status != 0)
486  {
487  rfbClientLog("Server refused VeNCrypt version %d.%d.\n", (int)major, (int)minor);
488  return FALSE;
489  }
490 
491  if (!ReadVeNCryptSecurityType(client, &authScheme)) return FALSE;
492  if (!ReadFromRFBServer(client, (char *)&status, 1) || status != 1)
493  {
494  rfbClientLog("Server refused VeNCrypt authentication %d (%d).\n", authScheme, (int)status);
495  return FALSE;
496  }
497  client->subAuthScheme = authScheme;
498 
499  /* Some VeNCrypt security types are anonymous TLS, others are X509 */
500  switch (authScheme)
501  {
502  case rfbVeNCryptTLSNone:
503  case rfbVeNCryptTLSVNC:
504  case rfbVeNCryptTLSPlain:
505 #ifdef LIBVNCSERVER_HAVE_SASL
506  case rfbVeNCryptTLSSASL:
507 #endif /* LIBVNCSERVER_HAVE_SASL */
508  anonTLS = TRUE;
509  break;
510  default:
511  anonTLS = FALSE;
512  break;
513  }
514 
515  /* Get X509 Credentials if it's not anonymous */
516  if (!anonTLS)
517  {
518  rfbCredential *cred;
519 
520  if (!client->GetCredential)
521  {
522  rfbClientLog("GetCredential callback is not set.\n");
523  return FALSE;
524  }
525  cred = client->GetCredential(client, rfbCredentialTypeX509);
526  if (!cred)
527  {
528  rfbClientLog("Reading credential failed\n");
529  return FALSE;
530  }
531 
532  x509_cred = CreateX509CertCredential(cred);
533  FreeX509Credential(cred);
534  if (!x509_cred) return FALSE;
535  }
536 
537  /* Start up the TLS session */
538  if (!InitializeTLSSession(client, anonTLS)) return FALSE;
539 
540  if (anonTLS)
541  {
542  if (!SetTLSAnonCredential(client)) return FALSE;
543  }
544  else
545  {
546  /* Set the certificate verification callback. */
547  gnutls_certificate_set_verify_function (x509_cred, verify_certificate_callback);
548  gnutls_session_set_ptr ((gnutls_session_t)client->tlsSession, (void *)client);
549 
550  if ((ret = gnutls_credentials_set((gnutls_session_t)client->tlsSession, GNUTLS_CRD_CERTIFICATE, x509_cred)) < 0)
551  {
552  rfbClientLog("Cannot set x509 credential: %s.\n", gnutls_strerror(ret));
553  FreeTLS(client);
554  return FALSE;
555  }
556  }
557 
558  if (!HandshakeTLS(client)) return FALSE;
559 
560  /* We are done here. The caller should continue with client->subAuthScheme
561  * to do actual sub authentication.
562  */
563  return TRUE;
564 }

566 int
567 ReadFromTLS(rfbClient* client, char *out, unsigned int n)
568 {
569  ssize_t ret;
570 
571  LOCK(client->tlsRwMutex);
572  ret = gnutls_record_recv((gnutls_session_t)client->tlsSession, out, n);
573  UNLOCK(client->tlsRwMutex);
574 
575  if (ret >= 0) return ret;
576  if (ret == GNUTLS_E_REHANDSHAKE || ret == GNUTLS_E_AGAIN)
577  {
578  errno = EAGAIN;
579  } else
580  {
581  rfbClientLog("Error reading from TLS: %s.\n", gnutls_strerror(ret));
582  errno = EINTR;
583  }
584  return -1;
585 }

587 int
588 WriteToTLS(rfbClient* client, const char *buf, unsigned int n)
589 {
590  unsigned int offset = 0;
591  ssize_t ret;
592 
593  while (offset < n)
594  {
595  LOCK(client->tlsRwMutex);
596  ret = gnutls_record_send((gnutls_session_t)client->tlsSession, buf+offset, (size_t)(n-offset));
597  UNLOCK(client->tlsRwMutex);
598 
599  if (ret == 0) continue;
600  if (ret < 0)
601  {
602  if (ret == GNUTLS_E_AGAIN || ret == GNUTLS_E_INTERRUPTED) continue;
603  rfbClientLog("Error writing to TLS: %s.\n", gnutls_strerror(ret));
604  return -1;
605  }
606  offset += (unsigned int)ret;
607  }
608 
609  return offset;
610 }
