112  * This method is called from InitFileTransfer method and
113  * if the command line option for ftproot is provided.

115 int
116 SetFtpRoot(char* path)
117 {
118  struct stat stat_buf;
119  DIR* dir = NULL;
120 
121  rfbLog("tightvnc-filetransfer/SetFtpRoot\n");
122 
123  if((path == NULL) || (strlen(path) == 0) || (strlen(path) > (PATH_MAX - 1))) {
124  rfbLog("File [%s]: Method [%s]: parameter passed is improper, ftproot"
125  " not changed\n", __FILE__, __FUNCTION__);
126  return FALSE;
127  }
128 
129  if(stat(path, &stat_buf) < 0) {
130  rfbLog("File [%s]: Method [%s]: Reading stat for file %s failed\n",
131  __FILE__, __FUNCTION__, path);
132  return FALSE;
133  }
134 
135  if(S_ISDIR(stat_buf.st_mode) == 0) {
136  rfbLog("File [%s]: Method [%s]: path specified is not a directory\n",
137  __FILE__, __FUNCTION__);
138  return FALSE;
139  }
140 
141  if((dir = opendir(path)) == NULL) {
142  rfbLog("File [%s]: Method [%s]: Not able to open the directory\n",
143  __FILE__, __FUNCTION__);
144  return FALSE;
145  }
146  else {
147  closedir(dir);
148  dir = NULL;
149  }
150 
151 
152  memset(ftproot, 0, PATH_MAX);
153  if(path[strlen(path)-1] == '/') {
154  memcpy(ftproot, path, strlen(path)-1);
155  }
156  else
157  memcpy(ftproot, path, strlen(path));
158 
159 
160  return TRUE;
161 }

164 /*
165  * Get the home directory for the user name
166  * param: username - name of the user for whom the home directory is required.
167  * returns: returns the home directory for the user, or null in case the entry
168  * is not found or any error. The returned string must be freed by calling the
169  * freehomedir function.
170 */
171 char*
172 GetHomeDir(uid_t uid)
173 {
174  char *homedir = NULL;
175 #ifdef WIN32
176  PWSTR profileDir = NULL;
177  if (SHGetKnownFolderPath(&FOLDERID_Profile, 0, NULL, &profileDir) != S_OK)
178  {
179  return NULL;
180  }
181 
182  int homedirlen = WideCharToMultiByte(CP_UTF8, 0, profileDir, -1, homedir, 0, NULL, NULL);
183  if (homedirlen<=0 || (homedir = malloc(homedirlen)) == NULL)
184  {
185  return NULL;
186  }
187  WideCharToMultiByte(CP_UTF8, 0, profileDir, -1, homedir, homedirlen, NULL, NULL);
188  CoTaskMemFree(profileDir);
189 #else
190  struct passwd *pwEnt = NULL;
191 
192  pwEnt = getpwuid (uid);
193  if (pwEnt == NULL)
194  return NULL;
195 
196  if(pwEnt->pw_dir != NULL) {
197  homedir = strdup (pwEnt->pw_dir);
198  }
199 #endif
200 
201  return homedir;
202 }

222  * When the console sends the File Transfer Request, it sends the file path with
223  * ftproot as "/". So on Agent, to get the absolute file path we need to prepend
224  * the ftproot to it.

228 {
229  char p[PATH_MAX];
230  memset(p, 0, PATH_MAX);
231 
232  if( (path == NULL) ||
233  (strlen(path) == 0) ||
234  (strlen(path)+strlen(ftproot) > PATH_MAX - 1) ) {
235 
236  rfbLog("File [%s]: Method [%s]: cannot create path for file transfer\n",
237  __FILE__, __FUNCTION__);
238  return NULL;
239  }
240 
241  memcpy(p, path, strlen(path));
242  memset(path, 0, PATH_MAX);
243  sprintf(path, "%s%s", ftproot, p);
244 
245  return path;
246 }

274 /*
275  * HandleFileListRequest method is called when the server receives
276  * FileListRequest. In case of success a file list is sent to the client.
277  * For File List Request there is no failure reason sent.So here in case of any
278  * "unexpected" error no information will be sent. As these conditions should
279  * never come. Lets hope it never arrives :)
280  * In case of dir open failure an empty list will be sent, just the header of
281  * the message filled up. So on console you will get an Empty listing.
282  */

287  int n = 0;

289  FileTransferMsg fileListMsg;
290 
291  memset(&msg, 0, sizeof(rfbClientToServerTightMsg));
292  memset(path, 0, PATH_MAX);
293  memset(&fileListMsg, 0, sizeof(FileTransferMsg));
294 
295  if(cl == NULL) {
296  rfbLog("File [%s]: Method [%s]: Unexpected error: rfbClientPtr is null\n",
297  __FILE__, __FUNCTION__);
298  return;
299  }
300 
301  if((n = rfbReadExact(cl, ((char *)&msg)+1, sz_rfbFileListRequestMsg-1)) <= 0) {
302 
303  if (n < 0)
304  rfbLog("File [%s]: Method [%s]: Socket error while reading dir name"
305  " length\n", __FILE__, __FUNCTION__);
306 
307  rfbCloseClient(cl);
308  return;
309  }
310 
311  msg.flr.dirNameSize = Swap16IfLE(msg.flr.dirNameSize);
312  if ((msg.flr.dirNameSize == 0) ||
313  (msg.flr.dirNameSize > (PATH_MAX - 1))) {
314 
315  rfbLog("File [%s]: Method [%s]: Unexpected error:: path length is "
316  "greater that PATH_MAX\n", __FILE__, __FUNCTION__);
317 
318  return;
319  }

321  if((n = rfbReadExact(cl, path, msg.flr.dirNameSize)) <= 0) {
322 
323  if (n < 0)
324  rfbLog("File [%s]: Method [%s]: Socket error while reading dir name\n",
325  __FILE__, __FUNCTION__);
326 
327  rfbCloseClient(cl);
328  return;
329  }
330 
331  if(ConvertPath(path) == NULL) {
332

334  rfbLog("File [%s]: Method [%s]: Unexpected error: path is NULL",
335  __FILE__, __FUNCTION__);
336  return;
337  }
338 
339  fileListMsg = GetFileListResponseMsg(path, (char) (msg.flr.flags));
340 
341  if((fileListMsg.data == NULL) || (fileListMsg.length == 0)) {
342 
343  rfbLog("File [%s]: Method [%s]: Unexpected error:: Data to be sent is "
344  "of Zero length\n", __FILE__, __FUNCTION__);
345  return;
346  }
347 
348  rfbWriteExact(cl, fileListMsg.data, fileListMsg.length);
349 
350  FreeFileTransferMsg(fileListMsg);

499 void*
500 RunFileDownloadThread(void* client)
501 {
502  rfbClientPtr cl = (rfbClientPtr) client;
503  rfbTightClientPtr rtcp = rfbGetTightClientData(cl);
504  FileTransferMsg fileDownloadMsg;
505 
506  if(rtcp == NULL)
507  return NULL;
508 
509  memset(&fileDownloadMsg, 0, sizeof(FileTransferMsg));
510  do {
511  pthread_mutex_lock(&fileDownloadMutex);
512  fileDownloadMsg = GetFileDownloadResponseMsgInBlocks(cl, rtcp);
513  pthread_mutex_unlock(&fileDownloadMutex);
514 
515  if((fileDownloadMsg.data != NULL) && (fileDownloadMsg.length != 0)) {
516  if(rfbWriteExact(cl, fileDownloadMsg.data, fileDownloadMsg.length) < 0) {
517  rfbLog("File [%s]: Method [%s]: Error while writing to socket \n"
518  , __FILE__, __FUNCTION__);
519  FreeFileTransferMsg(fileDownloadMsg);
520  return NULL;
521  }
522  FreeFileTransferMsg(fileDownloadMsg);
523  }
524  } while(rtcp->rcft.rcfd.downloadInProgress == TRUE);
525  return NULL;
526 }
