107 static void webSocketsGenSha1Key(char *target, int size, char *key)
108 {
109  struct iovec iov[2];
110  unsigned char hash[20];
111 
112  iov[0].iov_base = key;
113  iov[0].iov_len = strlen(key);
114  iov[1].iov_base = GUID;
115  iov[1].iov_len = sizeof(GUID) - 1;
116  digestsha1(iov, 2, hash);
117  if (-1 == rfbBase64NtoP(hash, sizeof(hash), target, size))
118  rfbErr("rfbBase64NtoP failed\n");

125 rfbBool
126 webSocketsCheck (rfbClientPtr cl)
127 {
128  char bbuf[4], *scheme;
129  int ret;
130 
131  ret = rfbPeekExactTimeout(cl, bbuf, 4,
132  WEBSOCKETS_CLIENT_CONNECT_WAIT_MS);
133  if ((ret < 0) && (errno == ETIMEDOUT)) {
134  rfbLog("Normal socket connection\n");
135  return TRUE;
136  } else if (ret <= 0) {
137  rfbErr("webSocketsHandshake: unknown connection error\n");
138  return FALSE;
139  }
140 
141  if (strncmp(bbuf, "RFB ", 4) == 0) {
142  rfbLog("Normal socket connection\n");
143  return TRUE;
144  } else if (strncmp(bbuf, "\x16", 1) == 0 || strncmp(bbuf, "\x80", 1) == 0) {
145  rfbLog("Got TLS/SSL WebSockets connection\n");
146  if (-1 == rfbssl_init(cl)) {
147  rfbErr("webSocketsHandshake: rfbssl_init failed\n");
148  return FALSE;
149  }
150  ret = rfbPeekExactTimeout(cl, bbuf, 4, WEBSOCKETS_CLIENT_CONNECT_WAIT_MS);
151  scheme = "wss";
152  } else {
153  scheme = "ws";
154  }
155 
156  if (strncmp(bbuf, "GET ", 4) != 0) {
157  rfbErr("webSocketsHandshake: invalid client header\n");
158  return FALSE;
159  }
160 
161  rfbLog("Got '%s' WebSockets handshake\n", scheme);
162 
163  if (!webSocketsHandshake(cl, scheme)) {
164  return FALSE;
165  }
166  /* Start WebSockets framing */
167  return TRUE;
168 }

170 static rfbBool
171 webSocketsHandshake(rfbClientPtr cl, char *scheme)
172 {
173  char *buf, *response, *line;
174  int n, linestart = 0, len = 0, llen, base64 = TRUE;
175  char prefix[5], trailer[17];
176  char *path = NULL, *host = NULL, *origin = NULL, *protocol = NULL;
177  char *key1 = NULL, *key2 = NULL, *key3 = NULL;
178  char *sec_ws_origin = NULL;
179  char *sec_ws_key = NULL;
180  char sec_ws_version = 0;
181  ws_ctx_t *wsctx = NULL;
182 
183  buf = (char *) malloc(WEBSOCKETS_MAX_HANDSHAKE_LEN);
184  if (!buf) {
185  rfbLogPerror("webSocketsHandshake: malloc");
186  return FALSE;
187  }
188  response = (char *) malloc(WEBSOCKETS_MAX_HANDSHAKE_LEN);
189  if (!response) {
190  free(buf);
191  rfbLogPerror("webSocketsHandshake: malloc");
192  return FALSE;
193  }
194 
195  while (len < WEBSOCKETS_MAX_HANDSHAKE_LEN-1) {
196  if ((n = rfbReadExactTimeout(cl, buf+len, 1,
197  WEBSOCKETS_CLIENT_SEND_WAIT_MS)) <= 0) {
198  if ((n < 0) && (errno == ETIMEDOUT)) {
199  break;
200  }
201  if (n == 0)
202  rfbLog("webSocketsHandshake: client gone\n");
203  else
204  rfbLogPerror("webSocketsHandshake: read");
205  free(response);
206  free(buf);
207  return FALSE;
208  }
209 
210  len += 1;
211  llen = len - linestart;
212  if (((llen >= 2)) && (buf[len-1] == '\n')) {
213  line = buf+linestart;
214  if ((llen == 2) && (strncmp("\r\n", line, 2) == 0)) {
215  if (key1 && key2) {
216  if ((n = rfbReadExact(cl, buf+len, 8)) <= 0) {
217  if ((n < 0) && (errno == ETIMEDOUT)) {
218  break;
219  }
220  if (n == 0)
221  rfbLog("webSocketsHandshake: client gone\n");
222  else
223  rfbLogPerror("webSocketsHandshake: read");
224  free(response);
225  free(buf);
226  return FALSE;
227  }
228  rfbLog("Got key3\n");
229  key3 = buf+len;
230  len += 8;
231  } else {
232  buf[len] = '\0';
233  }
234  break;
235  } else if ((llen >= 16) && ((strncmp("GET ", line, min(llen,4))) == 0)) {
236  /* 16 = 4 ("GET ") + 1 ("/.*") + 11 (" HTTP/1.1\r\n") */
237  path = line+4;
238  buf[len-11] = '\0'; /* Trim trailing " HTTP/1.1\r\n" */
239  cl->wspath = strdup(path);
240  /* rfbLog("Got path: %s\n", path); */
241  } else if ((strncasecmp("host: ", line, min(llen,6))) == 0) {
242  host = line+6;
243  buf[len-2] = '\0';
244  /* rfbLog("Got host: %s\n", host); */
245  } else if ((strncasecmp("origin: ", line, min(llen,8))) == 0) {
246  origin = line+8;
247  buf[len-2] = '\0';
248  /* rfbLog("Got origin: %s\n", origin); */
249  } else if ((strncasecmp("sec-websocket-key1: ", line, min(llen,20))) == 0) {
250  key1 = line+20;
251  buf[len-2] = '\0';
252  /* rfbLog("Got key1: %s\n", key1); */
253  } else if ((strncasecmp("sec-websocket-key2: ", line, min(llen,20))) == 0) {
254  key2 = line+20;
255  buf[len-2] = '\0';
256  /* rfbLog("Got key2: %s\n", key2); */
257  /* HyBI */
258 
259  } else if ((strncasecmp("sec-websocket-protocol: ", line, min(llen,24))) == 0) {
260  protocol = line+24;
261  buf[len-2] = '\0';
262  rfbLog("Got protocol: %s\n", protocol);
263  } else if ((strncasecmp("sec-websocket-origin: ", line, min(llen,22))) == 0) {
264  sec_ws_origin = line+22;
265  buf[len-2] = '\0';
266  } else if ((strncasecmp("sec-websocket-key: ", line, min(llen,19))) == 0) {
267  sec_ws_key = line+19;
268  buf[len-2] = '\0';
269  } else if ((strncasecmp("sec-websocket-version: ", line, min(llen,23))) == 0) {
270  sec_ws_version = strtol(line+23, NULL, 10);
271  buf[len-2] = '\0';
272  }
273 
274  linestart = len;
275  }
276  }
277 
278  /* older hixie handshake, this could be removed if
279  * a final standard is established -- removed now */
280  if (!sec_ws_version) {
281  rfbErr("Hixie no longer supported\n");
282  free(response);
283  free(buf);
284  return FALSE;
285  }
286 
287  if (!(path && host && (origin || sec_ws_origin))) {
288  rfbErr("webSocketsHandshake: incomplete client handshake\n");
289  free(response);
290  free(buf);
291  return FALSE;
292  }
293 
294  if ((protocol) && (strstr(protocol, "binary"))) {
295  rfbLog(" - webSocketsHandshake: using binary/raw encoding\n");
296  base64 = FALSE;
297  protocol = "binary";
298  } else {
299  rfbLog(" - webSocketsHandshake: using base64 encoding\n");
300  base64 = TRUE;
301  if ((protocol) && (strstr(protocol, "base64"))) {
302  protocol = "base64";
303  } else {
304  protocol = "";
305  }
306  }
307 
308  /*
309  * Generate the WebSockets server response based on the the headers sent
310  * by the client.
311  */
312  char accept[B64LEN(SHA1_HASH_SIZE) + 1];
313  rfbLog(" - WebSockets client version hybi-%02d\n", sec_ws_version);
314  webSocketsGenSha1Key(accept, sizeof(accept), sec_ws_key);
315 
316  if(strlen(protocol) > 0) {
317  len = snprintf(response, WEBSOCKETS_MAX_HANDSHAKE_LEN,
318  SERVER_HANDSHAKE_HYBI, accept, protocol);
319  } else {
320  len = snprintf(response, WEBSOCKETS_MAX_HANDSHAKE_LEN,
321  SERVER_HANDSHAKE_HYBI_NO_PROTOCOL, accept);
322  }
323 
324  if (rfbWriteExact(cl, response, len) < 0) {
325  rfbErr("webSocketsHandshake: failed sending WebSockets response\n");
326  free(response);
327  free(buf);
328  return FALSE;
329  }
330  /* rfbLog("webSocketsHandshake: %s\n", response); */
331  free(response);
332  free(buf);
333 
334  wsctx = calloc(1, sizeof(ws_ctx_t));
335  wsctx->encode = webSocketsEncodeHybi;
336  wsctx->decode = webSocketsDecodeHybi;
337  wsctx->ctxInfo.readFunc = ws_read;
338  wsctx->base64 = base64;
339  hybiDecodeCleanupComplete(wsctx);
340  cl->wsctx = (wsCtx *)wsctx;
341  return TRUE;
342 }

357 static int
358 webSocketsEncodeHybi(rfbClientPtr cl, const char *src, int len, char **dst)
359 {
360  int blen, ret = -1, sz = 0;
361  unsigned char opcode = '\0'; /* TODO: option! */
362  ws_header_t *header;
363  ws_ctx_t *wsctx = (ws_ctx_t *)cl->wsctx;
364 
365 
366  /* Optional opcode:
367  * 0x0 - continuation
368  * 0x1 - text frame (base64 encode buf)
369  * 0x2 - binary frame (use raw buf)
370  * 0x8 - connection close
371  * 0x9 - ping
372  * 0xA - pong
373  **/
374  if (!len) {
375  /* nothing to encode */
376  return 0;
377  }
378 
379  header = (ws_header_t *)wsctx->codeBufEncode;
380 
381  if (wsctx->base64) {
382  opcode = WS_OPCODE_TEXT_FRAME;
383  /* calculate the resulting size */
384  blen = B64LEN(len);
385  } else {
386  opcode = WS_OPCODE_BINARY_FRAME;
387  blen = len;
388  }
389 
390  header->b0 = 0x80 | (opcode & 0x0f);
391  if (blen <= 125) {
392  header->b1 = (uint8_t)blen;
393  sz = 2;
394  } else if (blen <= 65536) {
395  header->b1 = 0x7e;
396  header->u.s16.l16 = WS_HTON16((uint16_t)blen);
397  sz = 4;
398  } else {
399  header->b1 = 0x7f;
400  header->u.s64.l64 = WS_HTON64(blen);
401  sz = 10;
402  }
403 
404  if (wsctx->base64) {
405  if (-1 == (ret = rfbBase64NtoP((unsigned char *)src, len, wsctx->codeBufEncode + sz, sizeof(wsctx->codeBufEncode) - sz))) {
406  rfbErr("%s: Base 64 encode failed\n", __func__);
407  } else {
408  if (ret != blen)
409  rfbErr("%s: Base 64 encode; something weird happened\n", __func__);
410  ret += sz;
411  }
412  } else {
413  memcpy(wsctx->codeBufEncode + sz, src, len);
414  ret = sz + len;
415  }
416 
417  *dst = wsctx->codeBufEncode;
418 
419  return ret;
420 }
