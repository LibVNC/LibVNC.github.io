667 
668  if ((n = rfbReadExact(cl, pv, sz_rfbProtocolVersionMsg)) <= 0) {
669  if (n == 0)
670  rfbLog("rfbProcessClientProtocolVersion: client gone\n");
671  else
672  rfbLogPerror("rfbProcessClientProtocolVersion: read");
673  rfbCloseClient(cl);
674  return;
675  }

677  pv[sz_rfbProtocolVersionMsg] = 0;
678  if (sscanf(pv,rfbProtocolVersionFormat,&major_,&minor_) != 2) {
679  rfbErr("rfbProcessClientProtocolVersion: not a valid RFB client: %s\n", pv);
680  rfbCloseClient(cl);
681  return;
682  }
683  rfbLog("Client Protocol Version %d.%d\n", major_, minor_);
684 
685  if (major_ != rfbProtocolMajorVersion) {
686  rfbErr("RFB protocol version mismatch - server %d.%d, client %d.%d",
687  cl->screen->protocolMajorVersion, cl->screen->protocolMinorVersion,
688  major_,minor_);
689  rfbCloseClient(cl);
690  return;
691  }
692 
693  /* Check for the minor version use either of the two standard version of RFB */
694  /*
695  * UltraVNC Viewer detects FileTransfer compatible servers via rfb versions
696  * 3.4, 3.6, 3.14, 3.16
697  * It's a bad method, but it is what they use to enable features...
698  * maintaining RFB version compatibility across multiple servers is a pain
699  * Should use something like ServerIdentity encoding
700  */
701  cl->protocolMajorVersion = major_;
702  cl->protocolMinorVersion = minor_;
703 
704  rfbLog("Protocol version sent %d.%d, using %d.%d\n",
705  major_, minor_, rfbProtocolMajorVersion, cl->protocolMinorVersion);
706 
707  rfbAuthNewClient(cl);
